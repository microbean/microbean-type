017package org.microbean.type;
018
019import java.lang.reflect.Type;
020
021/**
022 * A {@link java.lang.reflect.WildcardType} implementation that has
023 * only {@linkplain java.lang.reflect.WildcardType#getUpperBounds()
024 * upper bounds}.
025 *
026 * @author <a href="https://about.me/lairdnelson"
027 * target="_parent">Laird Nelson</a>
028 *
029 * @see java.lang.reflect.WildcardType
030 *
031 * @see java.lang.reflect.WildcardType#getUpperBounds()
032 */
033public final class UpperBoundedWildcardType extends AbstractWildcardType {
034
035
036  /*
037   * Static fields.
038   */
039
040
041  private static final long serialVersionUID = 2L;
042
043
044  /*
045   * Constructors.
046   */
047
048
049  /**
050   * Creates a new {@link UpperBoundedWildcardType}.
051   *
052   * @param upperBound the sole {@linkplain
053   * java.lang.reflect.WildcardType#getUpperBounds() upper bound}; may
054   * be {@code null} in which case an array consisting solely of
055   * {@link Object Object.class} will be used instead
056   */
057  public UpperBoundedWildcardType(final Type upperBound) {
058    super(upperBound == null ? (Type[])null : new Type[] { upperBound }, null);
059  }
060
061  /**
062   * Creates a new {@link UpperBoundedWildcardType}.
063   *
064   * @param upperBounds the {@linkplain
065   * java.lang.reflect.WildcardType#getUpperBounds() upper bounds};
066   * may be {@code null}
067   */
068  public UpperBoundedWildcardType(final Type[] upperBounds) {
069    super(upperBounds, null);
070  }
071
072}
