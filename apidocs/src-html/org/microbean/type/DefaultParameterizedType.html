017package org.microbean.type;
018
019import java.io.IOException;
020import java.io.ObjectInputStream;
021import java.io.ObjectOutputStream;
022import java.io.Serializable;
023
024import java.lang.reflect.ParameterizedType;
025import java.lang.reflect.Type;
026
027import java.util.Arrays;
028import java.util.Objects;
029import java.util.StringJoiner;
030
031/**
032 * A {@link ParameterizedType} implementation.
033 *
034 * @author <a href="https://about.me/lairdnelson"
035 * target="_parent">Laird Nelson</a>
036 *
037 * @see ParameterizedType
038 */
039public final class DefaultParameterizedType extends AbstractType implements ParameterizedType {
040
041
042  /*
043   * Static fields.
044   */
045
046
047  private static final long serialVersionUID = 2L;
048
049
050  /*
051   * Instance fields.
052   */
053
054
055  private transient Type ownerType;
056
057  private transient Type rawType;
058
059  private transient Type[] actualTypeArguments;
060
061  private transient int hashCode;
062
063
064  /*
065   * Constructors.
066   */
067
068
069  /**
070   * Creates a new {@link DefaultParameterizedType}.
071   *
072   * @param ownerType the {@link Type} that owns this {@link
073   * DefaultParameterizedType}; may be (and usually is) {@code null};
074   * if non-{@code null} is most commonly a {@link Class} as in all
075   * JDKs through at least 13
076   *
077   * @param rawType the raw type; must not be {@code null}; is most
078   * commonly a {@link Class} as in all JDKs through at least 13
079   *
080   * @param actualTypeArguments the actual {@linkplain
081   * #getActualTypeArguments() actual type arguments} of this {@link
082   * DefaultParameterizedType}; may be {@code null}; will be cloned if
083   * non-{@code null}
084   *
085   * @exception NullPointerException if {@code rawType} is {@code
086   * null}
087   */
088  public DefaultParameterizedType(final Type ownerType, final Type rawType, final Type... actualTypeArguments) {
089    super();
090    this.ownerType = ownerType;
091    this.rawType = Objects.requireNonNull(rawType, "rawType");
092    if (actualTypeArguments == null || actualTypeArguments.length <= 0) {
093      this.actualTypeArguments = EMPTY_TYPE_ARRAY;
094    } else {
095      this.actualTypeArguments = actualTypeArguments.clone();
096    }
097    this.hashCode = this.computeHashCode();
098  }
099
100  /**
101   * Creates a new {@link DefaultParameterizedType}.
102   *
103   * @param other the {@link ParameterizedType} to copy information
104   * from; must not be {@code null}
105   *
106   * @exception NullPointerException if {@code other} is {@code null}
107   *
108   * @see #DefaultParameterizedType(Type, Type, Type...)
109   */
110  public DefaultParameterizedType(final ParameterizedType other) {
111    this(other.getOwnerType(), other.getRawType(), other.getActualTypeArguments());
112  }
113
114
115  /*
116   * Instance methods.
117   */
118
119
120  @Override
121  public final Type getOwnerType() {
122    return this.ownerType;
123  }
124
125  @Override
126  public final Type getRawType() {
127    return this.rawType;
128  }
129
130  @Override
131  public final Type[] getActualTypeArguments() {
132    return this.actualTypeArguments.clone();
133  }
134
135  @Override
136  public final int hashCode() {
137    return this.hashCode;
138  }
139
140  private final int computeHashCode() {
141    return Types.hashCode(this);
142  }
143
144  @Override
145  public final boolean equals(final Object other) {
146    if (other == this) {
147      return true;
148    } else if (other instanceof ParameterizedType) {
149      final ParameterizedType her = (ParameterizedType)other;
150      return
151        Objects.equals(this.ownerType, her.getOwnerType()) &&
152        Objects.equals(this.rawType, her.getRawType()) &&
153        Arrays.equals(this.actualTypeArguments, her.getActualTypeArguments());
154    } else {
155      return false;
156    }
157  }
158
159  @Override
160  public String toString() {
161    final StringBuilder sb = new StringBuilder();
162    final Type ownerType = this.ownerType;
163    if (ownerType == null) {
164      sb.append(this.rawType.getTypeName());
165    } else {
166      sb.append(ownerType.getTypeName()).append("$");
167      final Type rawType = this.rawType;
168      if (ownerType instanceof ParameterizedType) {
169        sb.append(rawType.getTypeName().replace(((ParameterizedType)ownerType).getRawType().getTypeName() + "$", ""));
170      } else if (rawType instanceof Class) {
171        sb.append(((Class<?>)rawType).getSimpleName());
172      } else {
173        sb.append(rawType.getTypeName());
174      }
175    }
176    final Type[] actualTypeArguments = this.actualTypeArguments;
177    if (actualTypeArguments != null && actualTypeArguments.length > 0) {
178      final StringJoiner stringJoiner = new StringJoiner(", ", "<", ">");
179      stringJoiner.setEmptyValue("");
180      for (final Type actualTypeArgument : actualTypeArguments) {
181        stringJoiner.add(actualTypeArgument.getTypeName());
182      }
183      sb.append(stringJoiner.toString());
184    }
185    return sb.toString();
186  }
187
188  private final void readObject(final ObjectInputStream stream) throws ClassNotFoundException, IOException {
189    stream.defaultReadObject();
190    final Object ownerType = stream.readObject();
191    if (ownerType == null) {
192      this.ownerType = null;
193    } else {
194      this.ownerType = (Type)ownerType;
195    }
196    final Object rawType = stream.readObject();
197    if (rawType == null) {
198      throw new IOException(new NullPointerException("rawType"));
199    } else {
200      this.rawType = (Type)rawType;
201    }
202    final Serializable[] serializableActualTypeArguments = (Serializable[])stream.readObject();
203    if (serializableActualTypeArguments == null || serializableActualTypeArguments.length <= 0) {
204      this.actualTypeArguments = EMPTY_TYPE_ARRAY;
205    } else {
206      this.actualTypeArguments = new Type[serializableActualTypeArguments.length];
207      System.arraycopy(serializableActualTypeArguments, 0, this.actualTypeArguments, 0, serializableActualTypeArguments.length);
208    }
209    this.hashCode = this.computeHashCode();
210  }
211
212  private final void writeObject(final ObjectOutputStream stream) throws IOException {
213    stream.defaultWriteObject();
214    stream.writeObject(Types.toSerializableType(this.ownerType));
215    stream.writeObject(Types.toSerializableType(this.rawType));
216    final Type[] actualTypeArguments = this.actualTypeArguments;
217    if (actualTypeArguments == null || actualTypeArguments.length <= 0) {
218      stream.writeObject(new Serializable[0]);
219    } else {
220      final Serializable[] newTypeArguments = new Serializable[actualTypeArguments.length];
221      for (int i = 0; i < newTypeArguments.length; i++) {
222        newTypeArguments[i] = Types.toSerializableType(actualTypeArguments[i]);
223      }
224      stream.writeObject(newTypeArguments);
225    }
226  }
227
228
229  /*
230   * Static methods.
231   */
232
233
234  public static final DefaultParameterizedType valueOf(final ParameterizedType type) {
235    if (type == null) {
236      return null;
237    } else if (type instanceof DefaultParameterizedType) {
238      return (DefaultParameterizedType)type;
239    } else {
240      return new DefaultParameterizedType(type);
241    }
242  }
