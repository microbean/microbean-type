017package org.microbean.type;
018
019import java.lang.reflect.ParameterizedType;
020import java.lang.reflect.Type;
021import java.lang.reflect.TypeVariable;
022
023import java.util.function.BiFunction;
024
025import org.microbean.development.annotation.Experimental;
026
027/**
028 * An abstract {@link TypeSemantics} that, in general, compares {@link
029 * Type}s using invariant semantics.
030 *
031 * <p>Normally {@link java.lang.reflect.WildcardType} instances would
032 * be compared using covariant semantics so often subclasses will
033 * override the {@link #getSemanticsFor(Type, Type)}
034 * appropriately.</p>
035 *
036 * @author <a href="https://about.me/lairdnelson"
037 * target="_parent">Laird Nelson</a>
038 *
039 * @see CovariantTypeSemantics
040 *
041 * @see TypeSemantics
042 */
043@Experimental
044public abstract class InvariantTypeSemantics extends TypeSemantics {
045
046  /**
047   * Creates a new {@link InvariantTypeSemantics}.
048   *
049   * @param box whether or not {@link Type}s will be {@linkplain
050   * Types#box(Type) boxed} before testing their assignability
051   *
052   * @see #isBoxing()
053   */
054  protected InvariantTypeSemantics(final boolean box) {
055    super(box);
056  }
057
058
059  /*
060   * Instance methods.
061   */
062
063  
064  @Override
065  protected final boolean isAssignable(final Class<?> receiverType,
066                                       final Class<?> payloadType) {
067    return this.box(receiverType).equals(this.box(payloadType));
068  }
069
070  @Override
071  protected final boolean isAssignable(final ParameterizedType receiverType,
072                                       final ParameterizedType payloadType) {
073    final boolean returnValue;
074    if (receiverType.getRawType().equals(payloadType.getRawType())) {
075      final Type[] receiverActualTypeArguments = receiverType.getActualTypeArguments();
076      final Type[] payloadActualTypeArguments = payloadType.getActualTypeArguments();
077      if (receiverActualTypeArguments.length == payloadActualTypeArguments.length) {
078        boolean temp = true;
079        for (int i = 0; i < receiverActualTypeArguments.length; i++) {
080          if (!this.isAssignable(receiverActualTypeArguments[i], payloadActualTypeArguments[i])) {
081            temp = false;
082            break;
083          }
084        }
085        returnValue = false;
086      } else {
087        returnValue = false;
088      }
089    } else {
090      returnValue = false;
091    }
092    return returnValue;
093  }
094
095  @Override
096  protected final boolean isAssignable(final TypeVariable<?> receiverType,
097                                       final TypeVariable<?> payloadType) {
098    return receiverType.equals(payloadType);
099  }
100
101}
