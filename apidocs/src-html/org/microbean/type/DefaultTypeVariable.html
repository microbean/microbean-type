039/**
040 * An {@link AbstractType} and a {@link TypeVariable} implementation.
041 *
042 * @param <T> the kind of {@link GenericDeclaration} that this {@link
043 * DefaultTypeVariable} was declared by
044 *
045 * @author <a href="https://about.me/lairdnelson"
046 * target="_parent">Laird Nelson</a>
047 */
048public final class DefaultTypeVariable<T extends GenericDeclaration> extends AbstractType implements TypeVariable<T> {

074  /**
075   * Creates a new {@link DefaultTypeVariable}.
076   *
077   * @param other a {@link TypeVariable} providing important locating
078   * information; must not be {@code null}
079   *
080   * @exception NullPointerException if {@code other} is {@code null}
081   */
082  public DefaultTypeVariable(final TypeVariable<? extends T> other) {
083    this(other.getGenericDeclaration(), tv -> tv.getName().equals(other.getName()));
084  }
085
086  /**
087   * Creates a new {@link DefaultTypeVariable}.
088   *
089   * @param genericDeclaration a {@link GenericDeclaration} whose
090   * {@linkplain GenericDeclaration#getTypeParameters() type
091   * variables} will be inspected; must not be {@code null}
092   *
093   * @param name the value that the return value of an invocation of
094   * {@link TypeVariable#getName()} must have when invoked on a member
095   * of the array returned by an invocation of {@link
096   * GenericDeclaration#getTypeParameters()} on the supplied {@code
097   * genericDeclaration}; must not be {@code null}
098   *
099   * @exception NullPointerException if {@code genericDeclaration} is
100   * {@code null}
101   *
102   * @exception IllegalArgumentException if the supplied {@code name}
103   * identifies no {@link TypeVariable}s
104   */
105  public DefaultTypeVariable(final T genericDeclaration, final String name) {
106    this(genericDeclaration, tv -> tv.getName().equals(name));
107  }
108
109  /**
110   * Creates a new {@link DefaultTypeVariable}.
111   *
112   * @param genericDeclaration a {@link GenericDeclaration} whose
113   * {@linkplain GenericDeclaration#getTypeParameters() type
114   * variables} will be inspected; must not be {@code null}
115   *
116   * @param predicate the {@link Predicate} that will identify the
117   * {@link TypeVariable} that this {@link DefaultTypeVariable} will
118   * wrap; must not be {@code null}
119   *
120   * @exception NullPointerException if either {@code
121   * genericDeclaration} or {@code predicate} is {@code null}
122   *
123   * @exception IllegalArgumentException if the supplied {@link
124   * Predicate} identifies no {@link TypeVariable}s
125   */
126  @SuppressWarnings("unchecked")
127  public DefaultTypeVariable(final T genericDeclaration, final Predicate<? super TypeVariable<?>> predicate) {
128    super();
129    for (final TypeVariable<?> tv : genericDeclaration.getTypeParameters()) {
130      if (predicate.test(tv)) {
131        this.delegate = (TypeVariable<? extends T>)tv;
132        break;
133      }
134    }
135    if (this.delegate == null) {
136      throw new IllegalArgumentException("genericDeclaration: " + genericDeclaration + "; predicate: " + predicate);
137    }
138    this.hashCode = this.computeHashCode();
139  }

244  @SuppressWarnings("unchecked")
245  private final void readObject(final ObjectInputStream stream) throws ClassNotFoundException, IOException {
246    stream.defaultReadObject();
247    final T gd;
248    final Object o = stream.readObject();
249    if (o instanceof Class) {
250      gd = (T)o;
251    } else if (o instanceof String) {
252      final String name = (String)o;
253      final Class<?>[] parameterTypes = (Class<?>[])stream.readObject();
254      final Class<?> declaringClass = (Class<?>)stream.readObject();
255      T temp = null;
256      try {
257        if ("<init>".equals(name)) {
258          temp = (T)declaringClass.getDeclaredConstructor(parameterTypes);
259        } else {
260          temp = (T)declaringClass.getDeclaredMethod(name, parameterTypes);
261        }
262      } catch (final ReflectiveOperationException reflectiveOperationException) {
263        throw new IOException(reflectiveOperationException.getMessage(), reflectiveOperationException);
264      } finally {
265        gd = temp;
266      }
267    } else {
268      throw new IllegalStateException();
269    }
270    final String name = (String)stream.readObject();
271    for (final TypeVariable<?> tv : gd.getTypeParameters()) {
272      if (tv.getName().equals(name)) {
273        this.delegate = (TypeVariable<? extends T>)tv;
274        break;
275      }
276    }
277    if (this.delegate == null) {
278      throw new IllegalStateException();
279    }
280    this.hashCode = this.computeHashCode();
281  }
282
283  private final void writeObject(final ObjectOutputStream stream) throws IOException {
284    stream.defaultWriteObject();
285    final GenericDeclaration gd = this.getGenericDeclaration();
286    if (gd instanceof Class) {
287      stream.writeObject(gd);
288    } else if (gd instanceof Executable) {
289      final Executable e = (Executable)gd;
290      stream.writeObject(e instanceof Constructor ? "<init>" : ((Method)e).getName());
291      stream.writeObject(e.getParameterTypes());
292      stream.writeObject(e.getDeclaringClass());
293    } else {
294      throw new AssertionError();
295    }
296    stream.writeObject(this.delegate.getName());
297  }

299  /**
300   * Returns a {@link DefaultTypeVariable} representing the supplied
301   * {@link TypeVariable}.
302   *
303   * <p>If the supplied {@link TypeVariable} is:</p>
304   *
305   * <ul>
306   *
307   * <li>{@code null}: {@code null} is returned</li>
308   *
309   * <li>an instance of {@link DefaultTypeVariable}: the supplied
310   * {@code type} is returned</li>
311   *
312   * <li>anything else: a new {@link DefaultTypeVariable} is returned</li>
313   *
314   * </ul>
315   *
316   * @param <T> the type of {@link GenericDeclaration} that declares
317   * the type variable
318   *
319   * @param type the {@link TypeVariable} in question; may be {@code
320   * null} in which case {@code null} will be returned
321   *
322   * @return a {@link DefaultTypeVariable}, or {@code null}
323   *
324   * @nullability This method may return {@code null}.
325   *
326   * @threadsafety This method is safe for concurrent use by multiple
327   * threads.
328   *
329   * @idempotency This method is idempotent and deterministic.
330   */
331  public static final <T extends GenericDeclaration> DefaultTypeVariable<T> valueOf(final TypeVariable<T> type) {
332    if (type == null) {
333      return null;
334    } else if (type instanceof DefaultTypeVariable) {
335      return (DefaultTypeVariable<T>)type;
336    } else {
337      return new DefaultTypeVariable<>(type);
338    }
339  }
