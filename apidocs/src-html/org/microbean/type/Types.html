103  /**
104   * For the vast majority of cases, returns the supplied {@link Type}
105   * unchanged, but in the case where a raw {@link Class} is supplied,
106   * returns an equivalent {@link ParameterizedType} whose {@linkplain
107   * ParameterizedType#getActualTypeArguments() actual type arguments}
108   * are not resolved.
109   *
110   * @param type the {@link Type} to normalize; may be {@code null} in
111   * which case {@code null} will be returned; will often be returned
112   * unchanged
113   *
114   * @return the normalized {@link Type}; {@code null} if {@code type}
115   * was {@code null}
116   *
117   * @nullability This method may return {@code null} only if the
118   * supplied {@link Type} is {@code null}.
119   *
120   * @threadsafety This method is safe for concurrent use by multiple
121   * threads.
122   */
123  public static final Type normalize(final Type type) {
124    if (type instanceof Class) {
125      final Class<?> cls = (Class<?>)type;
126      final Type[] typeParameters = cls.getTypeParameters();
127      if (typeParameters.length <= 0) {
128        final Type componentType = cls.getComponentType();
129        if (componentType == null) {
130          assert !isGeneric(cls);
131          assert !isRawType(cls);
132          return type;
133        } else {
134          final Type normalizedComponentType = normalize(componentType); // XXX recursive
135          return componentType == normalizedComponentType ? type : new DefaultGenericArrayType(normalizedComponentType);
136        }
137      } else {
138        assert !cls.isArray();
139        assert isGeneric(cls);
140        return new DefaultParameterizedType(cls.getDeclaringClass(), cls, typeParameters);
141      }
142    } else {
143      return type;
144    }
145  }

147  /**
148   * Returns {@code true} if and only if the supplied {@link Type} is
149   * a <em>raw type</em> according to <a
150   * href="https://docs.oracle.com/javase/specs/jls/se11/html/jls-4.html#jls-4.8"
151   * target="_parent">the rules of the Java Language
152   * Specification</a>.
153   *
154   * <p>This method returns {@code true} if:</p>
155   *
156   * <ul>
157   *
158   * <li>The supplied {@link Type} is an instance of {@link Class},
159   * <strong>and any</strong> of the following conditions is true:
160   *
161   * <ul>
162   *
163   * <li>The return value of invoking this method on the {@linkplain
164   * Class class}'s {@linkplain Class#getComponentType() component
165   * type} is {@code true}</li>
166   *
167   * <li>The {@linkplain Class class} {@linkplain
168   * Class#getTypeParameters() has type parameters}</li>
169   *
170   * </ul>
171   *
172   * </li>
173   *
174   * </ul>
175   *
176   * @param type the {@link Type} in question; may be {@code null} in
177   * which case {@code false} will be returned
178   *
179   * @return {@code true} if and only if the supplied {@link Type} is
180   * a raw {@link Class}
181   *
182   * @see <a
183   * href="https://docs.oracle.com/javase/specs/jls/se11/html/jls-4.html#jls-4.8"
184   * target="_parent">The Java Language Specification Section 4.8</a>
185   */
186  public static final boolean isRawType(final Type type) {
187    return type instanceof Class && isRawType((Class<?>)type);
188  }

190  private static final boolean isRawType(final Class<?> c) {
191    if (c == null) {
192      return false;
193    } else if (isGeneric(c) || isRawType(c.getComponentType())) {
194      return true;
195    } else {
196      // See https://docs.oracle.com/javase/specs/jls/se11/html/jls-4.html#jls-4.8
197      final Class<?> dc = c.getDeclaringClass();
198      // Don't recurse, because it's only the immediate parent that
199      // we're supposed to test
200      return dc != null && dc.getTypeParameters().length > 0;
201    }
202  }

205   * Returns {@code true} if and only if the supplied {@link Class} is
206   * <em>generic</em> according to <a
207   * href="https://docs.oracle.com/javase/specs/jls/se11/html/jls-8.html#jls-8.1.2"
208   * target="_parent">the rules of the Java Language
209   * Specification</a>.
210   *
211   * @param cls the {@link Class} in question; may be {@code null} in
212   * which case {@code false} will be returned
213   *
214   * @return {@code true} if the supplied {@link Class} is non-{@code
215   * null} and returns an array with more than zero elements from an
216   * invocation of its {@link Class#getTypeParameters()} method;
217   * {@code false} otherwise
218   *
219   * @see Class#getTypeParameters()
220   *
221   * @see <a
222   * href="https://docs.oracle.com/javase/specs/jls/se11/html/jls-8.html#jls-8.1.2"
223   * target="_parent">The Java Language Specification Section
224   * 8.1.2</a>
225   */
226  // https://docs.oracle.com/javase/specs/jls/se11/html/jls-8.html#jls-8.1.2
227  // "A class is generic if the class declaration declares one or more
228  // type variables (§4.4)."
229  public static final boolean isGeneric(final Class<?> cls) {
230    return cls != null && cls.getTypeParameters().length > 0;
231  }

268  /**
269   * Resolves the supplied {@link Type} and returns the result, or
270   * the {@link Type} itself if the {@link Type} could not be resolved.
271   *
272   * <p>Type resolution is delegated to the supplied {@link Function}
273   * which is most commonly a method reference to the {@link
274   * Map#get(Object)} method of a {@link Map Map&lt;Type, Type&gt;}.
275   * Generally speaking, however, the type resolver may take any
276   * side-effect-free action it wishes.</p>
277   *
278   * @param type the {@link Type} to resolve; may be {@code null}
279   *
280   * @param typeResolver the {@link Function} whose {@link
281   * Function#apply(Object)} method will be called on, among other
282   * things, {@link ParameterizedType} {@linkplain
283   * ParameterizedType#getActualTypeArguments() type arguments}; must
284   * not be {@code null}; may return {@code null}; will never be
285   * supplied with a {@code null} {@link Type}
286   *
287   * @return the resolved {@link Type}, or {@code null}
288   *
289   * @exception NullPointerException if {@code typeResolver} is {@code
290   * null}
291   *
292   * @nullability This method will return {@code null} only when the
293   * supplied {@link Type} is {@code null}.
294   *
295   * @threadsafety This method is safe for concurrent use by multiple
296   * threads.
297   *
298   * @idempotency This method is idempotent and deterministic.
299   */
300  public static final Type resolve(Type type,
301                                   final Function<? super Type, ? extends Type> typeResolver) {
302    if (type == null) {
303      return null;
304    } else if (type instanceof Class) {
305      return resolve((Class<?>)type, typeResolver);
306    } else if (type instanceof ParameterizedType) {
307      return resolve((ParameterizedType)type, typeResolver);
308    } else if (type instanceof GenericArrayType) {
309      return resolve((GenericArrayType)type, typeResolver);
310    } else if (type instanceof TypeVariable) {
311      return resolve((TypeVariable<?>)type, typeResolver);
312    } else if (type instanceof WildcardType) {
313      return resolve((WildcardType)type, typeResolver);
314    } else {
315      // (Unknown type, so we don't know how to resolve it.)
316      return type;
317    }
318  }
319
320  private static final Type resolve(final Class<?> type,
321                                    final Function<? super Type, ? extends Type> typeResolver) {
322    final Type candidate = typeResolver.apply(type);
323    return candidate == null ? type : candidate;
324  }

326  private static final Type resolve(final ParameterizedType type,
327                                    final Function<? super Type, ? extends Type> typeResolver) {
328    final Type candidate = typeResolver.apply(type);
329    if (candidate == null) {
330      final Type[] actualTypeArguments = type.getActualTypeArguments();
331      final int length = actualTypeArguments.length;
332      final Type[] resolvedActualTypeArguments = new Type[length];
333      boolean createNewType = false;
334      for (int i = 0; i < length; i++) {
335        final Type actualTypeArgument = actualTypeArguments[i];
336        final Type resolvedActualTypeArgument = resolve(actualTypeArgument, typeResolver); // XXX recursive
337        resolvedActualTypeArguments[i] = resolvedActualTypeArgument;
338        if (!createNewType && actualTypeArgument != resolvedActualTypeArgument) {
339          // If they're not the same object reference, then resolution
340          // did something, so we have to return a new type, not the
341          // one we were handed.
342          createNewType = true;
343        }
344      }
345      return
346        createNewType ? new DefaultParameterizedType(type.getOwnerType(), type.getRawType(), resolvedActualTypeArguments) : type;
347    } else {
348      return candidate;
349    }
350
351  }
352
353  private static final Type resolve(final GenericArrayType type,
354                                    final Function<? super Type, ? extends Type> typeResolver) {
355    final Type candidate = typeResolver.apply(type);
356    if (candidate == null) {
357      final Type genericComponentType = type.getGenericComponentType();
358      final Type resolvedComponentType = resolve(genericComponentType, typeResolver); // XXX recursive
359      if (resolvedComponentType == null || resolvedComponentType == genericComponentType) {
360        // Identity means that basically resolution was a no-op, so the
361        // genericComponentType was whatever it was, and so we are going
362        // to be a no-op as well.
363        return type;
364      } else if (resolvedComponentType instanceof Class) {
365        // This might happen when genericComponentType was a
366        // TypeVariable.  In this case, it might get resolved to a
367        // simple scalar (e.g. Integer.class).  Now we have,
368        // effectively, a GenericArrayType whose genericComponentType is
369        // just a plain class.  That's actually just an ordinary Java
370        // array, so "resolve" this type by returning its array
371        // equivalent.
372        return Array.newInstance((Class<?>)resolvedComponentType, 0).getClass();
373      } else {
374        // If we get here, we know that resolution actually did
375        // something, so return a new GenericArrayType implementation
376        // whose component type is the resolved version of the
377        // original's.
378        return new DefaultGenericArrayType(resolvedComponentType);
379      }
380    } else {
381      return candidate;
382    }
383  }

430  /**
431   * Returns a non-{@code null}, {@linkplain
432   * Collections#unmodifiableSet(Set) unmodifiable <code>Set</code>}
433   * of {@link Type}s, each element of which is a {@link Type} which
434   * any instance bearing the supplied {@link Type} will implement.
435   *
436   * <p>The {@link Set} that is returned is guaranteed to not contain
437   * {@link WildcardType} or {@link TypeVariable} instances.</p>
438   *
439   * <p>If the supplied {@link Predicate} returns {@code true} from an
440   * invocation of its {@link Predicate#test(Object)} method, then the
441   * {@link Type} that was passed to it will not be contained by the
442   * returned {@link TypeSet}.</p>
443   *
444   * @param type the {@link Type} whose type closure should be
445   * computed; may be {@code null}
446   *
447   * @param removalPredicate a {@link Predicate} used to selectively
448   * remove some {@link Type}s from the computed {@link TypeSet}; may
449   * be {@code null}
450   *
451   * @return a non-{@code null} {@link TypeSet}, filtered by the
452   * supplied {@link Predicate}
453   *
454   * @nullability This method never returns {@code null}.
455   *
456   * @threadsafety This method is safe for concurrent use by multiple
457   * threads.
458   *
459   * @idempotency This method is idempotent.
460   */
461  public static final TypeSet toTypes(final Type type, Predicate<? super Type> removalPredicate) {
462    final Map<Type, Type> resolvedTypes = new HashMap<>();
463    toTypes(type, isRawType(type), resolvedTypes);
464    resolvedTypes.keySet().removeIf(removalPredicate == null ?
465                                    Predicate.not(Types::isClass) :
466                                    k -> !isClass(k) || removalPredicate.test(resolvedTypes.get(k)));
467    return resolvedTypes.isEmpty() ? TypeSet.EMPTY_TYPESET : new TypeSet(resolvedTypes.values());
468  }

470  private static final void toTypes(final Type type,
471                                    final boolean noParameterizedTypes,
472                                    final Map<Type, Type> resolvedTypes) {
473    if (type == null) {
474      // Do nothing on purpose
475    } else if (type instanceof Class) {
476      toTypes((Class<?>)type, noParameterizedTypes, resolvedTypes);
477    } else if (type instanceof ParameterizedType) {
478      toTypes((ParameterizedType)type, noParameterizedTypes, resolvedTypes);
479    } else if (type instanceof GenericArrayType) {
480      toTypes((GenericArrayType)type, noParameterizedTypes, resolvedTypes);
481    } else if (type instanceof TypeVariable) {
482      toTypes((TypeVariable<?>)type, noParameterizedTypes, resolvedTypes);
483    } else if (type instanceof WildcardType) {
484      toTypes((WildcardType)type, noParameterizedTypes, resolvedTypes);
485    } else {
486      throw new IllegalArgumentException("Unexpected type: " + type);
487    }
488  }
489
490  private static final void toTypes(final Class<?> cls,
491                                    final boolean noParameterizedTypes,
492                                    final Map<Type, Type> resolvedTypes) {
493    resolvedTypes.put(cls, resolve(cls, resolvedTypes::get));
494    final Type superclass = noParameterizedTypes ? cls.getSuperclass() : cls.getGenericSuperclass();
495    if (superclass != null) {
496      toTypes(superclass, noParameterizedTypes || isRawType(superclass), resolvedTypes);
497    }
498    final Type[] interfaces = noParameterizedTypes ? cls.getInterfaces() : cls.getGenericInterfaces();
499    for (final Type iface : interfaces) {
500      toTypes(iface, noParameterizedTypes || isRawType(iface), resolvedTypes);
501    }
502  }

504  private static final void toTypes(final ParameterizedType parameterizedType,
505                                    final boolean noParameterizedTypes,
506                                    final Map<Type, Type> resolvedTypes) {
507    final Class<?> rawType = erase(parameterizedType);
508    if (rawType != null) {
509      if (!noParameterizedTypes) {
510        final TypeVariable<?>[] typeVariables = rawType.getTypeParameters();
511        final Type[] typeArguments = parameterizedType.getActualTypeArguments();
512        assert typeVariables.length == typeArguments.length;
513        for (int i = 0; i < typeVariables.length; i++) {
514          resolvedTypes.put(typeVariables[i], resolve(typeArguments[i], resolvedTypes::get));
515        }
516      }
517      resolvedTypes.put(rawType, resolve(parameterizedType, resolvedTypes::get));
518      toTypes(rawType, noParameterizedTypes, resolvedTypes);
519    }
520  }
521
522  private static final void toTypes(final GenericArrayType genericArrayType,
523                                    final boolean noParameterizedTypes,
524                                    final Map<Type, Type> resolvedTypes) {
525    final Class<?> rawType = erase(genericArrayType);
526    if (rawType != null) {
527      assert rawType.isArray() : "Not an array type; check erase(GenericArrayType): " + rawType;
528      resolvedTypes.put(rawType, genericArrayType);
529      toTypes(rawType, noParameterizedTypes, resolvedTypes);
530    }
531  }

555   *
556   * @param <T> the kind of {@link Type} in question
557   *
558   * @param type the {@link Type} to test
559   *
560   * @return the result of <a
561   * href="https://docs.oracle.com/javase/specs/jls/se13/html/jls-5.html#jls-5.1.7">boxing
562   * conversion</a> on the supplied {@link Type}
563   *
564   * @nullability This method will return {@code null} if {@code type}
565   * is {@code null}.
566   *
567   * @threadsafety This method is safe for concurrent use by multiple
568   * threads.
569   *
570   * @idempotency This method is idempotent and deterministic.
571   *
572   * @see #isPrimitive(Type)
573   *
574   * @see <a
575   * href="https://docs.oracle.com/javase/specs/jls/se11/html/jls-5.html#jls-5.1.7">The
576   * Java Language Specification Section 5.1.7</a>
577   */
578  @SuppressWarnings("unchecked")
579  public static final <T extends Type> T box(final T type) {
580    return
581      type instanceof Class && ((Class<?>)type).isPrimitive() ?
582      (T)wrapperTypes.getOrDefault(type, (Class<?>)type) :
583      type;
584  }

667   * {@link #erase(Type)} <strong>on its {@linkplain

669   * returned.  <strong>Any other bounds are ignored.</strong></li>
670   *
671   * </ul>

681   *
682   * @threadsafety This method is safe for concurrent use by mutltiple
683   * threads.
684   *
685   * @idempotency This method is idempotent and deterministic.

763    final Type[] bounds = type.getBounds();
764    if (bounds == null || bounds.length <= 0) {

770      } else {
771        return new Type[] { soleBound };
772      }
773    } else {
774      return bounds.clone();

1075  private static final String toString(final ParameterizedType ptype) {
1076    if (ptype == null) {
1077      return "null";
1078    } else {
1079      final StringBuilder sb = new StringBuilder();
1080      final Type ownerType = ptype.getOwnerType();
1081      if (ownerType == null) {
1082        sb.append(toString(ptype.getRawType()));
1083      } else {
1084        sb.append(toString(ownerType)).append("$");
1085        final Type rawType = ptype.getRawType();
1086        if (ownerType instanceof ParameterizedType) {
1087          sb.append(toString(rawType).replace(toString(((ParameterizedType)ownerType).getRawType()) + "$", ""));
1088        } else if (rawType instanceof Class) {
1089          sb.append(((Class<?>)rawType).getSimpleName());
1090        } else {
1091          sb.append(toString(rawType));
1092        }
1093      }
1094      final Type[] actualTypeArguments = ptype.getActualTypeArguments();
1095      if (actualTypeArguments != null && actualTypeArguments.length > 0) {
1096        final StringJoiner stringJoiner = new StringJoiner(", ", "<", ">");
1097        stringJoiner.setEmptyValue("");
1098        for (final Type actualTypeArgument : actualTypeArguments) {
1099          stringJoiner.add(toString(actualTypeArgument));
1100        }
1101        sb.append(stringJoiner.toString());
1102      }
1103      return sb.toString();
1104    }

1119  private static final String toString(final WildcardType wc) {
1120    if (wc == null) {
1121      return "null";
1122    } else {
1123      final StringBuilder sb = new StringBuilder("?");
1124      Type[] bounds = wc.getLowerBounds();
1125      if (bounds == null || bounds.length <= 0) {
1126        // Upper bounds only.
1127        bounds = wc.getUpperBounds();
1128        if (bounds == null || bounds.length <= 0 || Object.class.equals(bounds[0])) {
1129          bounds = null;
1130        } else {
1131          sb.append(" extends ");
1132        }
1133      } else {
1134        // Lower bounds only.
1135        sb.append(" super ");
1136      }
1137      if (bounds != null) {
1138        assert bounds.length > 0;
1139        final StringJoiner sj = new StringJoiner(" & ");
1140        for (final Type bound: bounds) {
1141          sj.add(String.valueOf(toString(bound)));
1142        }
1143        sb.append(sj.toString());
1144      }
1145      return sb.toString();
1146    }
