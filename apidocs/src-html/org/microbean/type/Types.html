097  /**
098   * For the vast majority of cases, returns the supplied {@link Type}
099   * unchanged, but in the case where a raw {@link Class} is supplied,
100   * returns an equivalent {@link ParameterizedType} whose {@linkplain
101   * ParameterizedType#getActualTypeArguments() actual type arguments}
102   * are not resolved.
103   *
104   * @param type the {@link Type} to normalize; may be {@code null} in
105   * which case {@code null} will be returned; will often be returned
106   * unchanged
107   *
108   * @return the normalized {@link Type}; {@code null} if {@code type}
109   * was {@code null}
110   *
111   * @nullability This method may return {@code null} only if the
112   * supplied {@link Type} is {@code null}.
113   *
114   * @threadsafety This method is safe for concurrent use by multiple
115   * threads.
116   */
117  public static final Type normalize(final Type type) {
118    final Type returnValue;
119    if (type instanceof Class) {
120      final Class<?> cls = (Class<?>)type;
121      final Type[] typeParameters = cls.getTypeParameters();
122      if (typeParameters == null || typeParameters.length <= 0) {
123        final Type componentType = cls.getComponentType();
124        if (componentType == null) {
125          returnValue = type;
126        } else {
127          final Type normalizedComponentType = normalize(componentType); // XXX recursive
128          if (componentType == normalizedComponentType) {
129            returnValue = type;
130          } else {
131            returnValue = new DefaultGenericArrayType(normalizedComponentType);
132          }
133        }
134      } else {
135        returnValue = new DefaultParameterizedType(cls.getDeclaringClass(), cls, typeParameters);
136      }
137    } else {
138      returnValue = type;
139    }
140    return returnValue;
141  }

173  /**
174   * Resolves the supplied {@link Type} and returns the result.
175   *
176   * @param type the {@link Type} to resolve; may be {@code null}
177   *
178   * @param typeResolver the {@link Function} whose {@link
179   * Function#apply(Object)} method will be called on, among other
180   * things, {@link ParameterizedType} {@linkplain
181   * ParameterizedType#getActualTypeArguments() type arguments}; must
182   * not be {@code null}
183   *
184   * @param normalize whether the {@link #normalize(Type)} method
185   * should be called on the supplied {@link Type}
186   *
187   * @return the resolved {@link Type}, or {@code null}
188   *
189   * @nullability This method may return {@code null}.
190   *
191   * @threadsafety This method is safe for concurrent use by multiple
192   * threads.
193   *
194   * @idempotency This method is idempotent and deterministic.
195   *
196   * @since 0.0.3
197   */
198  public static final Type resolve(Type type, final Function<? super Type, ? extends Type> typeResolver, final boolean normalize) {
199    final Type returnValue;
200    if (normalize) {
201      type = normalize(type);
202    }
203    if (type instanceof Class) {
204      returnValue = resolve((Class<?>)type, typeResolver, normalize);
205    } else if (type instanceof ParameterizedType) {
206      returnValue = resolve((ParameterizedType)type, typeResolver, normalize);
207    } else if (type instanceof GenericArrayType) {
208      returnValue = resolve((GenericArrayType)type, typeResolver, normalize);
209    } else if (type instanceof TypeVariable) {
210      returnValue = resolve((TypeVariable<?>)type, typeResolver, normalize);
211    } else if (type instanceof WildcardType) {
212      // (No-op.)
213      returnValue = resolve((WildcardType)type, typeResolver, normalize);
214    } else {
215      // (Unknown type, so we don't know how to resolve it.)
216      returnValue = type;
217    }
218    return returnValue;
219  }

222    final Type candidate = typeResolver.apply(type);
223    return candidate == null ? type : candidate;

226  private static final Type resolve(final ParameterizedType type, final Function<? super Type, ? extends Type> typeResolver, final boolean normalize) {
227    final Type candidate = typeResolver.apply(type);
228    if (candidate == null) {
229      final Type[] actualTypeArguments = type.getActualTypeArguments();
230      final int length = actualTypeArguments.length;
231      final Type[] resolvedActualTypeArguments = new Type[length];
232      boolean createNewType = false;
233      for (int i = 0; i < length; i++) {
234        final Type actualTypeArgument = actualTypeArguments[i];
235        final Type resolvedActualTypeArgument = resolve(actualTypeArgument, typeResolver, normalize); // XXX recursive
236        resolvedActualTypeArguments[i] = resolvedActualTypeArgument;
237        if (actualTypeArgument != resolvedActualTypeArgument) {
238          // If they're not the same object reference, then resolution
239          // did something, so we have to return a new type, not the
240          // one we were handed.
241          createNewType = true;
242        }
243      }
244      return createNewType ? new DefaultParameterizedType(type.getOwnerType(), type.getRawType(), resolvedActualTypeArguments) : type;
245    } else {
246      return candidate;
247    }
248
249  }

251  private static final Type resolve(final GenericArrayType type, final Function<? super Type, ? extends Type> typeResolver, final boolean normalize) {
252    final Type returnValue;
253    final Type candidate = typeResolver.apply(type);
254    if (candidate == null) {
255      final Type genericComponentType = type.getGenericComponentType();
256      final Type resolvedComponentType = resolve(genericComponentType, typeResolver, normalize); // XXX recursive
257      assert resolvedComponentType != null;
258      if (resolvedComponentType == genericComponentType) {
259        // Identity means that basically resolution was a no-op, so the
260        // genericComponentType was whatever it was, and so we are going
261        // to be a no-op as well.
262        returnValue = type;
263      } else if (resolvedComponentType instanceof Class) {
264        // This might happen when genericComponentType was a
265        // TypeVariable.  In this case, it might get resolved to a
266        // simple scalar (e.g. Integer.class).  Now we have,
267        // effectively, a GenericArrayType whose genericComponentType is
268        // just a plain class.  That's actually just an ordinary Java
269        // array, so "resolve" this type by returning its array
270        // equivalent.
271        returnValue = Array.newInstance((Class<?>)resolvedComponentType, 0).getClass();
272      } else {
273        // If we get here, we know that resolution actually did
274        // something, so return a new GenericArrayType implementation
275        // whose component type is the resolved version of the
276        // original's.
277        returnValue = new DefaultGenericArrayType(resolvedComponentType);
278      }
279    } else {
280      returnValue = candidate;
281    }
282    return returnValue;
283  }

424   *
425   * @param normalize whether the {@link #normalize(Type)} method
426   * should be called on the supplied {@link Type} (provided it
427   * {@linkplain #isRawClass(Type) is not a raw <code>Class</code>})
428   *
429   * @return a non-{@code null} {@link TypeSet} <code>Set</code>} of
430   * {@link Type}s, filtered by the supplied {@link Predicate}
431   *
432   * @nullability This method never returns {@code null}.
433   *
434   * @threadsafety This method is safe for concurrent use by multiple
435   * threads.
436   *
437   * @idempotency This method is idempotent.
438   *
439   * @since 0.0.3
440   */
441  public static final TypeSet toTypes(final Type type, Predicate<? super Type> removalPredicate, final boolean normalize) {
442    final Map<Type, Type> resolvedTypes = new HashMap<>();
443    toTypes(type, isRawClass(type), resolvedTypes, normalize);
444    resolvedTypes.keySet().removeIf(removalPredicate == null ?
445                                    Predicate.not(Types::isClass) :
446                                    k -> !isClass(k) || removalPredicate.test(resolvedTypes.get(k)));
447    return resolvedTypes.isEmpty() ? TypeSet.EMPTY_TYPESET : new TypeSet(resolvedTypes.values());
448  }

450  private static final void toTypes(Type type,
451                                    final boolean noParameterizedTypes,
452                                    final Map<Type, Type> resolvedTypes,
453                                    final boolean normalize) {
454    if (normalize) {
455      type = normalize(type);
456    }
457    if (type == null) {
458      // Do nothing on purpose
459    } else if (type instanceof Class) {
460      toTypes((Class<?>)type, noParameterizedTypes, resolvedTypes, normalize);
461    } else if (type instanceof ParameterizedType) {
462      toTypes((ParameterizedType)type, noParameterizedTypes, resolvedTypes, normalize);
463    } else if (type instanceof GenericArrayType) {
464      toTypes((GenericArrayType)type, noParameterizedTypes, resolvedTypes, normalize);
465    } else if (type instanceof TypeVariable) {
466      // Do nothing on purpose
467    } else if (type instanceof WildcardType) {
468      // Do nothing on purpose
469    } else {
470      throw new IllegalArgumentException("Unexpected type: " + type);
471    }
472  }

474  private static final void toTypes(final Class<?> cls,
475                                    final boolean noParameterizedTypes,
476                                    final Map<Type, Type> resolvedTypes,
477                                    final boolean normalize) {
478    resolvedTypes.put(cls, resolve(cls, resolvedTypes::get, normalize));
479    final Type superclass = noParameterizedTypes ? cls.getSuperclass() : cls.getGenericSuperclass();
480    if (superclass != null) {
481      toTypes(superclass, noParameterizedTypes || isRawClass(superclass), resolvedTypes, normalize);
482    }
483    final Type[] interfaces = noParameterizedTypes ? cls.getInterfaces() : cls.getGenericInterfaces();
484    for (final Type iface : interfaces) {
485      toTypes(iface, noParameterizedTypes || isRawClass(iface), resolvedTypes, normalize);
486    }
487  }

489  private static final void toTypes(final ParameterizedType parameterizedType,
490                                    final boolean noParameterizedTypes,
491                                    final Map<Type, Type> resolvedTypes,
492                                    final boolean normalize) {
493    final Class<?> rawType = toClass(parameterizedType);
494    if (rawType != null) {
495      if (!noParameterizedTypes) {
496        final TypeVariable<?>[] typeVariables = rawType.getTypeParameters();
497        final Type[] typeVariableValues = parameterizedType.getActualTypeArguments();
498        assert typeVariables.length == typeVariableValues.length;
499        for (int i = 0; i < typeVariables.length; i++) {
500          resolvedTypes.put(typeVariables[i], resolve(typeVariableValues[i], resolvedTypes::get, normalize));
501        }
502      }
503      resolvedTypes.put(rawType, resolve(parameterizedType, resolvedTypes::get, normalize));
504      toTypes(rawType, noParameterizedTypes, resolvedTypes, normalize);
505    }
506  }
507
508  private static final void toTypes(final GenericArrayType genericArrayType,
509                                    final boolean noParameterizedTypes,
510                                    final Map<Type, Type> resolvedTypes,
511                                    final boolean normalize) {
512    final Class<?> rawType = toClass(genericArrayType);
513    if (rawType != null) {
514      final Class<?> arrayType = Array.newInstance(rawType, 0).getClass();
515      resolvedTypes.put(arrayType, genericArrayType);
516      toTypes(arrayType, noParameterizedTypes, resolvedTypes, normalize);
517    }
518  }

520  /**
521   * Returns the result of <a
522   * href="https://docs.oracle.com/javase/specs/jls/se13/html/jls-5.html#jls-5.1.7">boxing
523   * conversion</a> on the supplied {@link Type}.
524   *
525   * <p>In most cases this means the supplied {@link Type} is returned
526   * unchanged.</p>
527   *
528   * <p>This method calls {@link #isPrimitive(Type)} as part of its
529   * execution.</p>
530   *
531   * @param <T> the kind of {@link Type} in question
532   *
533   * @param type the {@link Type} to test
534   *
535   * @return the result of <a
536   * href="https://docs.oracle.com/javase/specs/jls/se13/html/jls-5.html#jls-5.1.7">boxing
537   * conversion</a> on the supplied {@link Type}
538   *
539   * @nullability This method will return {@code null} if {@code type}
540   * is {@code null}.
541   *
542   * @threadsafety This method is safe for concurrent use by multiple
543   * threads.
544   *
545   * @idempotency This method is idempotent and deterministic.
546   *
547   * @see #isPrimitive(Type)
548   */
549  @SuppressWarnings("unchecked")
550  public static final <T extends Type> T box(final T type) {
551    // We don't just blindly look up in wrapperTypes.  We call
552    // isPrimitive(Type) first (which will almost certainly be inlined
553    // by the JVM) to potentially avoid the Map lookup.
554    return isPrimitive(type) ? (T)wrapperTypes.getOrDefault(type, (Class<?>)type) : type;
555  }

639  /**
640   * Does what is necessary to extract a {@link Class} from the
641   * supplied {@link Type} if at all possible.
642   *
643   * <p><strong>Please read closely</strong> if you intend to supply
644   * either a {@link TypeVariable} or a {@link WildcardType} as an
645   * argument.</p>
646   *
647   * <ul>
648   *
649   * <li>If {@code null} is supplied, {@code null} is returned.</li>
650   *
651   * <li>If a {@link Class} is supplied, the {@link Class} is simply
652   * returned.</li>
653   *
654   * <li>If a {@link ParameterizedType} is supplied, the result of
655   * invoking {@link #toClass(Type)} on its {@linkplain
656   * ParameterizedType#getRawType() raw type} is returned.</li>
657   *
658   * <li>If a {@link GenericArrayType} is supplied, the result of
659   * invoking {@link Array#newInstance(Class, int)} with the return
660   * value of an invocation of{@link #toClass(Type)} on its
661   * {@linkplain GenericArrayType#getGenericComponentType() generic
662   * component type} and {@code 0} as its arguments is returned.</li>
663   *
664   * <li>If a {@link TypeVariable} is supplied, the result of invoking
665   * {@link #toClass(Type)} <strong>on its {@linkplain
666   * TypeVariable#getBounds() first bound}</strong> is returned (if it
667   * has one) or {@link Object Object.class} if it does not.
668   * <strong>Any other bounds are ignored.</strong></li>
669   *
670   * <li>If a {@link WildcardType} is supplied, the result of invoking
671   * {@link #toClass(Type)} <strong>on its {@linkplain
672   * WildcardType#getUpperBounds() first upper bound}</strong> is
673   * returned.  <strong>Any other bounds are ignored.</strong></li>
674   *
675   * </ul>
676   *
677   * @param type the {@link Type} for which a {@link Class} is to be
678   * returned; may be {@code null} in which case {@code null} will be
679   * returned
680   *
681   * @return a {@link Class}, or {@code null} if a suitable raw type
682   * could not be determined
683   *
684   * @nullability This method may return {@code null}.
685   *
686   * @threadsafety This method is safe for concurrent use by mutltiple
687   * threads.
688   *
689   * @idempotency This method is idempotent and deterministic.
690   */
691  public static final Class<?> toClass(final Type type) {
692    final Class<?> returnValue;
693    if (type == null) {
694      returnValue = null;
695    } else if (type instanceof Class) {
696      returnValue = toClass((Class<?>)type);
697    } else if (type instanceof ParameterizedType) {
698      returnValue = toClass((ParameterizedType)type);
699    } else if (type instanceof GenericArrayType) {
700      returnValue = toClass((GenericArrayType)type);
701    } else if (type instanceof TypeVariable) {
702      returnValue = toClass((TypeVariable<?>)type);
703    } else if (type instanceof WildcardType) {
704      returnValue = toClass((WildcardType)type);
705    } else {
706      returnValue = null;
707    }
708    return returnValue;
709  }
710
711  private static final Class<?> toClass(final Class<?> type) {
712    return type;
713  }
714
715  private static final Class<?> toClass(final ParameterizedType type) {
716    return type == null ? null : toClass(type.getRawType());
717  }
718
719  private static final Class<?> toClass(final GenericArrayType type) {
720    final Class<?> candidate = type == null ? null : toClass(type.getGenericComponentType());
721    return candidate == null ? null : (Class<?>)Array.newInstance(candidate, 0).getClass();
722  }
723
724  private static final Class<?> toClass(final TypeVariable<?> type) {
725    final Type[] bounds = type == null ? null : type.getBounds();
726    return bounds != null && bounds.length > 0 ? toClass(bounds[0]) : Object.class;
727  }
728
729  private static final Class<?> toClass(final WildcardType type) {
730    final Type[] bounds = type == null ? null : type.getUpperBounds();
731    return bounds != null && bounds.length > 0 ? toClass(bounds[0]) : Object.class;
732  }
733
734  public static final Serializable toSerializableType(final Type type) {
735    final Serializable returnValue;
736    if (type == null || type instanceof Serializable) {
737      returnValue = (Serializable)type;
738    } else if (type instanceof ParameterizedType) {
739      returnValue = new DefaultParameterizedType((ParameterizedType)type);
740    } else if (type instanceof GenericArrayType) {
741      returnValue = DefaultGenericArrayType.valueOf((GenericArrayType)type);
742    } else if (type instanceof TypeVariable) {
743      returnValue = new PartiallyImplementedTypeVariable((TypeVariable<?>)type);
744    } else if (type instanceof WildcardType) {
745      returnValue = new AbstractWildcardType((WildcardType)type);
746    } else {
747      throw new IllegalArgumentException("Unexpected type: " + type);
748    }
749    return returnValue;
750  }
