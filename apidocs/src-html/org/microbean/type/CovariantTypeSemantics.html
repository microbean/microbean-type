017package org.microbean.type;
018
019import java.lang.reflect.GenericArrayType;
020import java.lang.reflect.ParameterizedType;
021import java.lang.reflect.Type;
022import java.lang.reflect.TypeVariable;
023import java.lang.reflect.WildcardType;
024
025import java.util.Objects;
026
027import org.microbean.development.annotation.Experimental;
028
029/**
030 * A {@link TypeSemantics} specialization that implements the <a
031 * href="https://docs.oracle.com/javase/specs/jls/se14/html/jls-4.html">Java
032 * Language Specification type assignability rules</a>.
033 *
034 * @author <a href="https://about.me/lairdnelson"
035 * target="_parent">Laird Nelson</a>
036 */
037@Experimental
038public final class CovariantTypeSemantics extends TypeSemantics {
039
040
041  /*
042   * Instance fields.
043   */
044
045  
046  private final InvariantTypeSemantics invariantTypeSemantics;
047
048
049  /*
050   * Constructors.
051   */
052
053
054  /**
055   * Creates a new {@link CovariantTypeSemantics} instance.
056   *
057   * @param box whether or not {@link Type}s will be {@linkplain
058   * Types#box(Type) boxed} before testing their assignability
059   *
060   * @see #isBoxing()
061   */
062  public CovariantTypeSemantics(final boolean box) {
063    super(box);
064    this.invariantTypeSemantics = new InvariantTypeSemantics(box) {
065        @Override
066        protected final TypeSemantics getSemanticsFor(final Type receiverType, final Type payloadType) {
067          return receiverType instanceof WildcardType || payloadType instanceof WildcardType ? CovariantTypeSemantics.this : null;
068        }
069      };
070  }
071
072
073  /*
074   * Instance methods.
075   */
076  
077
078  /**
079   * Returns a {@link TypeSemantics} used for certain cases where
080   * invariant type assignability semantics are called for, such as
081   * when comparing {@link ParameterizedType} type arguments.
082   *
083   * @return a {@link TypeSemantics} used for certain cases where
084   * invariant type assignability semantics are called for; never
085   * {@code null}
086   *
087   * @nullability This method never returns {@code null}.
088   *
089   * @see InvariantTypeSemantics
090   */
091  public final TypeSemantics getInvariantTypeSemantics() {
092    return this.invariantTypeSemantics;
093  }
094
095  @Override
096  protected boolean isAssignable(final Class<?> receiverType,
097                                 final Class<?> payloadType) {
098    return this.box(receiverType).isAssignableFrom(this.box(payloadType));
099  }
100
101  @Override
102  protected boolean isAssignable(final Class<?> receiverType,
103                                 final ParameterizedType payloadType) {
104    return this.isAssignable(receiverType, payloadType.getRawType());
105  }
106
107  @Override
108  protected boolean isAssignable(final Class<?> receiverType,
109                                 final GenericArrayType payloadType) {
110    return
111      receiverType.equals(Object.class) ||
112      this.isAssignable(receiverType.getComponentType(), payloadType.getGenericComponentType());
113  }
114
115  @Override
116  protected boolean isAssignable(final Class<?> receiverType,
117                                 final TypeVariable<?> payloadType) {
118    boolean returnValue = false;
119    for (final Type payloadTypeBound : payloadType.getBounds()) {
120      if (this.isAssignable(receiverType, payloadTypeBound)) {
121        returnValue = true;
122        break;
123      }
124    }
125    return returnValue;
126  }
127
128  @Override
129  protected boolean isAssignable(final ParameterizedType receiverType,
130                                 final Class<?> payloadType) {
131    return
132      this.isAssignable(receiverType.getRawType(), payloadType) &&
133      (!Types.normalize(payloadType).equals(payloadType) || // TODO: won't this ALWAYS return false?
134       this.isAssignable0(receiverType, Types.toTypes(payloadType)));
135  }
136
137  @Override
138  protected boolean isAssignable(final ParameterizedType receiverType,
139                                 final ParameterizedType payloadType) {
140    return
141      this.isAssignable0(receiverType, payloadType) ||
142      this.isAssignable0(receiverType, Types.toTypes(payloadType));
143  }
144
145  private final boolean isAssignable0(final ParameterizedType receiverType,
146                                      final TypeSet payloadTypeSet) {
147    boolean returnValue = false;
148    for (final Type payloadType : payloadTypeSet) {
149      if (payloadType instanceof ParameterizedType && this.isAssignable0(receiverType, (ParameterizedType)payloadType)) {
150        returnValue = true;
151        break;
152      }
153    }
154    return returnValue;
155  }
156
157  private final boolean isAssignable0(final ParameterizedType receiverType,
158                                      final ParameterizedType payloadType) {
159    final boolean returnValue;
160    if (receiverType.getRawType().equals(payloadType.getRawType())) {
161      final Type[] receiverTypeTypeArguments = receiverType.getActualTypeArguments();
162      final Type[] payloadTypeTypeArguments = payloadType.getActualTypeArguments();
163      if (receiverTypeTypeArguments.length == payloadTypeTypeArguments.length) {
164        boolean temp = true;
165        for (int i = 0; i < receiverTypeTypeArguments.length; i++) {
166          if (!this.getInvariantTypeSemantics().isAssignable(receiverTypeTypeArguments[i], payloadTypeTypeArguments[i])) {
167            temp = false;
168            break;
169          }
170        }
171        returnValue = temp;
172      } else {
173        returnValue = false;
174      }
175    } else {
176      returnValue = false;
177    }
178    return returnValue;
179  }
180
181  @Override
182  protected boolean isAssignable(final ParameterizedType receiverType,
183                                 final TypeVariable<?> payloadType) {
184    boolean returnValue = false;
185    for (final Type payloadTypeBound : payloadType.getBounds()) {
186      if (this.isAssignable(receiverType, payloadTypeBound)) {
187        returnValue = true;
188        break;
189      }
190    }
191    return returnValue;
192  }
193
194  @Override
195  protected boolean isAssignable(final GenericArrayType receiverType,
196                                 final Class<?> payloadType) {
197    return this.isAssignable(receiverType.getGenericComponentType(), payloadType.getComponentType());
198  }
199
200  @Override
201  protected boolean isAssignable(final TypeVariable<?> receiverType,
202                                 final TypeVariable<?> payloadType) {
203    final boolean returnValue;
204    if (receiverType.equals(payloadType)) {
205      returnValue = true;
206    } else {
207      // (Note that the Java Language Specification guarantees that if
208      // a type variable extends another type variable, then the
209      // extended type variable will be its sole bound.)
210      final Object solePayloadTypeBound = payloadType.getBounds()[0];
211      returnValue =
212        solePayloadTypeBound instanceof TypeVariable &&
213        this.isAssignable(receiverType, (TypeVariable<?>)solePayloadTypeBound);
214    }
215    return returnValue;
216    
217  }
218
219  @Override
220  protected final boolean isAssignable(final WildcardType receiverType,
221                                       final Class<?> payloadType) {
222    return this.isAssignable0(receiverType, payloadType);
223  }
224
225  @Override
226  protected final boolean isAssignable(final WildcardType receiverType,
227                                       final ParameterizedType payloadType) {
228    return this.isAssignable0(receiverType, payloadType);
229  }
230
231  @Override
232  protected final boolean isAssignable(final WildcardType receiverType,
233                                       final GenericArrayType payloadType) {
234    return this.isAssignable0(receiverType, payloadType);
235  }
236
237  private final boolean isAssignable0(final WildcardType receiverType,
238                                      final Type payloadActualType) {
239    final boolean returnValue;
240    if (this.isAssignable(receiverType.getUpperBounds()[0], payloadActualType)) {
241      final Type[] receiverLowerBounds = receiverType.getLowerBounds();
242      // If there are lower bounds, then there will be only one.  See
243      // https://stackoverflow.com/a/6645454/208288 and
244      // https://github.com/openjdk/jdk/blob/d3d29a4f828053f5961bfd4f4ccd200791a7eeb1/src/java.base/share/classes/java/lang/reflect/WildcardType.java#L80-L82
245      returnValue =
246        receiverLowerBounds.length <= 0 || this.isAssignable(payloadActualType, receiverLowerBounds[0]);
247    } else {
248      returnValue = false;
249    }
250    return returnValue;
251  }
252
253  @Override
254  protected final boolean isAssignable(final WildcardType receiverType,
255                                       final TypeVariable<?> payloadType) {
256    final Type[] receiverLowerBounds = receiverType.getLowerBounds();
257    return
258      (receiverLowerBounds != null && receiverLowerBounds.length > 0) ?
259      // Why only the first bound? See:
260      // https://stackoverflow.com/a/6645454/208288
261      // https://github.com/openjdk/jdk/blob/d3d29a4f828053f5961bfd4f4ccd200791a7eeb1/src/java.base/share/classes/java/lang/reflect/WildcardType.java#L80-L82
262      this.isAssignable(payloadType, receiverLowerBounds[0]) :
263      // The Java Language Specification guarantees that if there
264      // aren't lower bounds, then there are upper bounds.  There's
265      // also only one; see
266      // https://docs.oracle.com/en/java/javase/14/docs/api/java.base/java/lang/reflect/WildcardType.html#getUpperBounds()
267      this.isAssignable(receiverType.getUpperBounds()[0], payloadType);
268  }
269
270  @Override
271  protected final boolean isAssignable(final WildcardType receiverType,
272                                       final WildcardType payloadType) {
273    final boolean returnValue;
274    // There are always upper bounds, and in fact only one; see
275    // https://docs.oracle.com/en/java/javase/14/docs/api/java.base/java/lang/reflect/WildcardType.html#getUpperBounds()
276    final Type receiverUpperBound = receiverType.getUpperBounds()[0];
277    if (this.isAssignable(receiverUpperBound, payloadType.getUpperBounds()[0])) {
278      final Type[] receiverLowerBounds = receiverType.getLowerBounds();
279      if (receiverLowerBounds.length > 0) {
280        final Type[] payloadLowerBounds = payloadType.getLowerBounds();
281        returnValue = payloadLowerBounds.length > 0 && this.isAssignable(payloadLowerBounds[0], receiverLowerBounds[0]);
282      } else {
283        returnValue = payloadType.getLowerBounds().length <= 0 || receiverUpperBound.equals(Object.class);
284      }
285    } else {
286      returnValue = false;
287    }
288    return returnValue;
289  }
