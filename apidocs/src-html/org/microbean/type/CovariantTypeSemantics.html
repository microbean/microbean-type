017package org.microbean.type;
018
019import java.lang.reflect.GenericArrayType;
020import java.lang.reflect.ParameterizedType;
021import java.lang.reflect.Type;
022import java.lang.reflect.TypeVariable;
023import java.lang.reflect.WildcardType;
024
025import org.microbean.development.annotation.Experimental;
026
027/**
028 * A {@link TypeSemantics} specialization that implements the <a
029 * href="https://docs.oracle.com/javase/specs/jls/se14/html/jls-4.html">Java
030 * Language Specification type assignability rules</a>.
031 *
032 * @author <a href="https://about.me/lairdnelson"
033 * target="_parent">Laird Nelson</a>
034 */
035@Experimental
036public final class CovariantTypeSemantics extends TypeSemantics {
037
038
039  /*
040   * Instance fields.
041   */
042
043
044  private final InvariantTypeSemantics invariantTypeSemantics;
045
046
047  /*
048   * Constructors.
049   */
050
051
052  /**
053   * Creates a new {@link CovariantTypeSemantics} instance.
054   *
055   * @param box whether or not {@link Type}s will be {@linkplain
056   * Types#box(Type) boxed} before testing their assignability
057   *
058   * @see #isBoxing()
059   */
060  public CovariantTypeSemantics(final boolean box) {
061    super(box);
062    this.invariantTypeSemantics = new InvariantTypeSemantics(box) {
063        @Override
064        protected final TypeSemantics getSemanticsFor(final Type receiverType, final Type payloadType) {
065          // Wildcards are always compared covariantly.
066          if (receiverType instanceof WildcardType || payloadType instanceof WildcardType) {
067            return CovariantTypeSemantics.this;
068          } else {
069            return super.getSemanticsFor(receiverType, payloadType);
070          }
071        }
072      };
073  }
074
075
076  /*
077   * Instance methods.
078   */
079
080
081  /**
082   * Returns a {@link TypeSemantics} used for certain cases where
083   * invariant type assignability semantics are called for, such as
084   * when comparing {@link ParameterizedType} type arguments.
085   *
086   * <p>It is worth noting that {@link WildcardType}s should always be
087   * compared covariantly and the {@link TypeSemantics} returned by
088   * this method is guaranteed to do so.</p>
089   *
090   * @return a {@link TypeSemantics} used for certain cases where
091   * invariant type assignability semantics are called for; never
092   * {@code null}
093   *
094   * @nullability This method never returns {@code null}.
095   *
096   * @see #isAssignable(ParameterizedType, ParameterizedType)
097   *
098   * @see InvariantTypeSemantics
099   */
100  public final TypeSemantics getInvariantTypeSemantics() {
101    return this.invariantTypeSemantics;
102  }
103
104  @Override
105  protected boolean isAssignable(final Class<?> receiverType,
106                                 final Class<?> payloadType) {
107    return this.box(receiverType).isAssignableFrom(this.box(payloadType));
108  }
109
110  @Override
111  protected boolean isAssignable(final Class<?> receiverType,
112                                 final ParameterizedType payloadType) {
113    return this.isAssignable(receiverType, payloadType.getRawType());
114  }
115
116  @Override
117  protected boolean isAssignable(final Class<?> receiverType,
118                                 final GenericArrayType payloadType) {
119    return
120      receiverType == Object.class ||
121      this.isAssignable(receiverType.getComponentType(), payloadType.getGenericComponentType());
122  }
123
124  @Override
125  protected boolean isAssignable(final Class<?> receiverType,
126                                 final TypeVariable<?> payloadType) {
127    for (final Type payloadTypeBound : payloadType.getBounds()) {
128      if (this.isAssignable(receiverType, payloadTypeBound)) {
129        return true;
130      }
131    }
132    return false;
133  }
134
135  @Override
136  protected boolean isAssignable(final ParameterizedType receiverType,
137                                 final Class<?> payloadType) {
138    return
139      this.isAssignable(receiverType.getRawType(), payloadType) &&
140      (!Types.normalize(payloadType).equals(payloadType) ||
141       this.isAssignable0(receiverType, Types.toTypes(payloadType)));
142  }
143
144  @Override
145  protected boolean isAssignable(final ParameterizedType receiverType,
146                                 final ParameterizedType payloadType) {
147    return this.isAssignable0(receiverType, Types.toTypes(payloadType));
148  }
149
150  private final boolean isAssignable0(final ParameterizedType receiverType,
151                                      final Iterable<? extends Type> payloadTypeSet) {
152    for (final Type payloadType : payloadTypeSet) {
153      if (payloadType instanceof ParameterizedType && this.isAssignable0(receiverType, (ParameterizedType)payloadType)) {
154        return true;
155      }
156    }
157    return false;
158  }
159
160  private final boolean isAssignable0(final ParameterizedType receiverType,
161                                      final ParameterizedType payloadType) {
162    if (receiverType.getRawType().equals(payloadType.getRawType())) {
163      final Type[] receiverTypeTypeArguments = receiverType.getActualTypeArguments();
164      final Type[] payloadTypeTypeArguments = payloadType.getActualTypeArguments();
165      if (receiverTypeTypeArguments.length == payloadTypeTypeArguments.length) {
166        // The TypeSemantics used here are invariant except for wildcards.
167        final TypeSemantics invariantTypeSemantics = this.getInvariantTypeSemantics();
168        for (int i = 0; i < receiverTypeTypeArguments.length; i++) {
169          if (!invariantTypeSemantics.isAssignable(receiverTypeTypeArguments[i], payloadTypeTypeArguments[i])) {
170            return false;
171          }
172        }
173        return true;
174      }
175    }
176    return false;
177  }
178
179  @Override
180  protected boolean isAssignable(final ParameterizedType receiverType,
181                                 final TypeVariable<?> payloadType) {
182    for (final Type payloadTypeBound : payloadType.getBounds()) {
183      if (this.isAssignable(receiverType, payloadTypeBound)) {
184        return true;
185      }
186    }
187    return false;
188  }
189
190  @Override
191  protected boolean isAssignable(final GenericArrayType receiverType,
192                                 final Class<?> payloadType) {
193    return this.isAssignable(receiverType.getGenericComponentType(), payloadType.getComponentType());
194  }
195
196  @Override
197  protected boolean isAssignable(final TypeVariable<?> receiverType,
198                                 final TypeVariable<?> payloadType) {
199    final boolean returnValue;
200    if (receiverType.equals(payloadType)) {
201      returnValue = true;
202    } else {
203      // (Note that the Java Language Specification guarantees that if
204      // a type variable extends another type variable, then the
205      // extended type variable will be its sole bound.)
206      final Object solePayloadTypeBound = payloadType.getBounds()[0];
207      returnValue =
208        solePayloadTypeBound instanceof TypeVariable && this.isAssignable(receiverType, (TypeVariable<?>)solePayloadTypeBound);
209    }
210    return returnValue;
211  }
212
213  @Override
214  protected final boolean isAssignable(final WildcardType receiverType,
215                                       final Class<?> payloadType) {
216    return this.isAssignable0(receiverType, payloadType);
217  }
218
219  @Override
220  protected final boolean isAssignable(final WildcardType receiverType,
221                                       final ParameterizedType payloadType) {
222    return this.isAssignable0(receiverType, payloadType);
223  }
224
225  @Override
226  protected final boolean isAssignable(final WildcardType receiverType,
227                                       final GenericArrayType payloadType) {
228    return this.isAssignable0(receiverType, payloadType);
229  }
230
231  private final boolean isAssignable0(final WildcardType receiverType,
232                                      final Type payloadActualType) {
233    if (this.isAssignable(receiverType.getUpperBounds()[0], payloadActualType)) {
234      final Type[] receiverLowerBounds = receiverType.getLowerBounds();
235      // If there are lower bounds, then there will be only one.  See
236      // https://stackoverflow.com/a/6645454/208288 and
237      // https://github.com/openjdk/jdk/blob/d3d29a4f828053f5961bfd4f4ccd200791a7eeb1/src/java.base/share/classes/java/lang/reflect/WildcardType.java#L80-L82
238      return receiverLowerBounds.length <= 0 || this.isAssignable(payloadActualType, receiverLowerBounds[0]);
239    } else {
240      return false;
241    }
242  }
243
244  @Override
245  protected final boolean isAssignable(final WildcardType receiverType,
246                                       final TypeVariable<?> payloadType) {
247    final Type[] receiverLowerBounds = receiverType.getLowerBounds();
248    return
249      (receiverLowerBounds != null && receiverLowerBounds.length > 0) ?
250      // Why only the first bound? See:
251      // https://stackoverflow.com/a/6645454/208288
252      // https://github.com/openjdk/jdk/blob/d3d29a4f828053f5961bfd4f4ccd200791a7eeb1/src/java.base/share/classes/java/lang/reflect/WildcardType.java#L80-L82
253      this.isAssignable(payloadType, receiverLowerBounds[0]) :
254      // The Java Language Specification guarantees that if there
255      // aren't lower bounds, then there are upper bounds.  There's
256      // also only one; see
257      // https://docs.oracle.com/en/java/javase/14/docs/api/java.base/java/lang/reflect/WildcardType.html#getUpperBounds()
258      this.isAssignable(receiverType.getUpperBounds()[0], payloadType);
259  }
260
261  @Override
262  protected final boolean isAssignable(final WildcardType receiverType,
263                                       final WildcardType payloadType) {
264    // There are always upper bounds, and in fact only one; see
265    // https://docs.oracle.com/en/java/javase/14/docs/api/java.base/java/lang/reflect/WildcardType.html#getUpperBounds()
266    final Type receiverUpperBound = receiverType.getUpperBounds()[0];
267    if (this.isAssignable(receiverUpperBound, payloadType.getUpperBounds()[0])) {
268      final Type[] receiverLowerBounds = receiverType.getLowerBounds();
269      if (receiverLowerBounds.length > 0) {
270        final Type[] payloadLowerBounds = payloadType.getLowerBounds();
271        return payloadLowerBounds.length > 0 && this.isAssignable(payloadLowerBounds[0], receiverLowerBounds[0]);
272      } else {
273        return payloadType.getLowerBounds().length <= 0 || receiverUpperBound.equals(Object.class);
274      }
275    }
276    return false;
277  }
