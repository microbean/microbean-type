031/**
032 * An {@linkplain Collections#unmodifiableSet(Set) unmodifiable}
033 * {@link AbstractSet} of {@link Type}s with additional convenient
034 * functionality.
035 *
036 * <p>{@linkplain #iterator() Iteration} over any {@link TypeSet}
037 * instance is guaranteed to be in some deterministic but otherwise
038 * unspecified order.  The order is subject to change between versions
039 * of this class.</p>
040 *
041 * @author <a href="https://about.me/lairdnelson"
042 * target="_parent">Laird Nelson</a>
043 */
044public class TypeSet extends AbstractSet<Type> {

093  /**
094   * Creates a {@link TypeSet} that {@linkplain #contains(Object)
095   * contains} only the supplied {@link Type} unless the supplied
096   * {@link Type} is {@code null}, in which case the resulting {@link
097   * TypeSet} will be forever {@linkplain #isEmpty() empty}.
098   *
099   * @param type the {@link Type} the new {@link TypeSet} will
100   * contain; may be {@code null} in which case the new {@link
101   * TypeSet} will be {@linkplain #isEmpty() empty}
102   */
103  public TypeSet(final Type type) {
104    this(type == null ? Collections.emptySet() : Collections.singleton(type));
105  }
106
107  /**
108   * Creates a {@link TypeSet} that {@linkplain #contains(Object)
109   * contains} only the {@link Type}s supplied unless the supplied
110   * {@link Collection} is {@code null} or {@linkplain
111   * Collection#isEmpty() empty}, in which case the resulting {@link
112   * TypeSet} will be forever {@linkplain #isEmpty() empty}.
113   *
114   * @param types the {@link Type}s the new {@link TypeSet} will
115   * contain; may be {@code null} or {@linkplain Collection#isEmpty()
116   * empty} in which case the new {@link TypeSet} will be forever
117   * {@linkplain #isEmpty() empty}
118   */
119  public TypeSet(final Collection<? extends Type> types) {
120    super();
121    if (types == null || types.isEmpty()) {
122      this.typeSet = Collections.emptySortedSet();
123    } else {
124      final SortedSet<Type> sortedTypeSet = new TreeSet<>(Types.typeNameComparator);
125      sortedTypeSet.addAll(types);
126      this.typeSet = Collections.unmodifiableSortedSet(sortedTypeSet);
127    }
128  }

136  /**
137   * Returns a non-{@code null}, {@linkplain
138   * Collections#unmodifiableSet(Set) unmodifiable} {@link SortedSet}
139   * of all raw types contained directly or indirectly by this {@link
140   * TypeSet}.
141   *
142   * <p>This method calls the {@link #getClasses()} and {@link
143   * #getInterfaces()} methods as part of its operation.</p>
144   *
145   * @return a non-{@code null}, {@linkplain
146   * Collections#unmodifiableSet(Set) unmodifiable} {@link SortedSet}
147   * of all raw types contained directly or indirectly by this {@link
148   * TypeSet}
149   *
150   * @see #getClasses()
151   *
152   * @see #getInterfaces()
153   *
154   * @nullability This method never returns {@code null}.
155   *
156   * @threadsafety This method is safe for concurrent use by multiple
157   * threads.
158   *
159   * @idempotency This method is idempotent and deterministic.  The
160   * {@link Set} that is returned consists of elements in a
161   * particular, but deliberately unspecified, order that is not
162   * dependent upon the ordering of the {@link Set} of {@link Type}s
163   * initially supplied to this {@link TypeSet}.
164   */
165  public final SortedSet<Class<?>> getRawTypes() {
166    SortedSet<Class<?>> rawTypes = this.rawTypes;
167    if (rawTypes == null) {
168      final SortedSet<Class<?>> classes = this.getClasses();
169      final SortedSet<Class<?>> interfaces = this.getInterfaces();
170      if (classes.isEmpty()) {
171        if (interfaces.isEmpty()) {
172          rawTypes = Collections.emptySortedSet();
173        } else {
174          rawTypes = interfaces;
175        }
176      } else if (interfaces.isEmpty()) {
177        rawTypes = classes;
178      } else {
179        final SortedSet<Class<?>> target = new TreeSet<>(Types.typeNameComparator);
180        target.addAll(classes);
181        target.addAll(interfaces);
182        rawTypes = Collections.unmodifiableSortedSet(target);
183      }
184      this.rawTypes = rawTypes;
185    }
186    return rawTypes;
187  }

189  /**
190   * Returns a non-{@code null}, {@linkplain
191   * Collections#unmodifiableSet(Set) unmodifiable} {@link SortedSet}
192   * of all {@link Class}es (not interfaces) contained directly or
193   * indirectly by this {@link TypeSet}.
194   *
195   * @return a non-{@code null}, {@linkplain
196   * Collections#unmodifiableSet(Set) unmodifiable} {@link Set} of all
197   * {@link Class}es (not interfaces) contained directly or indirectly
198   * by this {@link TypeSet}
199   *
200   * @nullability This method never returns {@code null}.
201   *
202   * @threadsafety This method is safe for concurrent use by multiple
203   * threads.
204   *
205   * @idempotency This method is idempotent and deterministic.  The
206   * {@link Set} that is returned consists of elements in a
207   * particular, but deliberately unspecified, order that is not
208   * dependent upon the ordering of the {@link Set} of {@link Type}s
209   * initially supplied to this {@link TypeSet}.
210   */
211  public final SortedSet<Class<?>> getClasses() {
212    SortedSet<Class<?>> classes = this.classes;
213    if (classes == null) {
214      if (this.isEmpty()) {
215        classes = Collections.emptySortedSet();
216      } else {
217        classes = new TreeSet<>(Types.typeNameComparator);
218        for (final Type type : this) {
219          // TODO: shouldn't we ignore any Type that is not either a Class or a ParameterizedType?
220          Class<?> rawType = Types.toClass(type);
221          while (rawType != null) {
222            if (!rawType.isInterface()) {
223              classes.add(rawType);
224            }
225            rawType = rawType.getSuperclass();
226          }
227        }
228        if (classes.isEmpty()) {
229          classes = Collections.emptySortedSet();
230        } else {
231          classes = Collections.unmodifiableSortedSet(classes);
232        }
233      }
234      assert classes != null;
235      this.classes = classes;
236    }
237    return classes;
238  }

240  /**
241   * Returns a non-{@code null}, {@linkplain
242   * Collections#unmodifiableSet(Set) unmodifiable} {@link SortedSet}
243   * of all {@link Type}s representing classes (not interfaces)
244   * contained directly or indirectly by this {@link TypeSet}.
245   *
246   * <p>{@link Type}s contained directly or indirectly by this {@link
247   * TypeSet} that are not instances of either {@link Class} or {@link
248   * ParameterizedType} are deliberately ignored.</p>
249   *
250   * @return a non-{@code null}, {@linkplain
251   * Collections#unmodifiableSet(Set) unmodifiable} {@link Set} of all
252   * {@link Type}s representing classes (not interfaces) contained
253   * directly or indirectly by this {@link TypeSet}
254   *
255   * @nullability This method never returns {@code null}.
256   *
257   * @threadsafety This method is safe for concurrent use by multiple
258   * threads.
259   *
260   * @idempotency This method is idempotent and deterministic.  The
261   * {@link Set} that is returned consists of elements in a
262   * particular, but deliberately unspecified, order that is not
263   * dependent upon the ordering of the {@link Set} of {@link Type}s
264   * initially supplied to this {@link TypeSet}.
265   */
266  public final SortedSet<Type> getGenericClasses() {
267    SortedSet<Type> genericClasses = this.genericClasses;
268    if (genericClasses == null) {
269      if (this.isEmpty()) {
270        genericClasses = Collections.emptySortedSet();
271      } else {
272        genericClasses = new TreeSet<>(Types.typeNameComparator);
273        for (Type type : this) {
274          while (type != null) {
275            type = addClass(type, genericClasses);
276          }
277        }
278        if (genericClasses.isEmpty()) {
279          genericClasses = Collections.emptySortedSet();
280        } else {
281          genericClasses = Collections.unmodifiableSortedSet(genericClasses);
282        }
283      }
284      assert genericClasses != null;
285      this.genericClasses = genericClasses;
286    }
287    return genericClasses;
288  }

291   * Returns a non-{@code null}, {@linkplain
292   * Collections#unmodifiableSet(Set) unmodifiable} {@link Set} of all
293   * interfaces contained directly or indirectly by this {@link
294   * TypeSet}.
295   *
296   * @return a non-{@code null}, {@linkplain
297   * Collections#unmodifiableSet(Set) unmodifiable} {@link Set} of all
298   * interfaces contained directly or indirectly by this {@link
299   * TypeSet}
300   *
301   * @nullability This method never returns {@code null}.
302   *
303   * @threadsafety This method is safe for concurrent use by multiple
304   * threads.
305   *
306   * @idempotency This method is idempotent and deterministic.  The
307   * {@link Set} that is returned consists of elements in a
308   * particular, but deliberately unspecified, order that is not
309   * dependent upon the ordering of the {@link Set} of {@link Type}s
310   * initially supplied to this {@link TypeSet}.
311   */
312  public final SortedSet<Class<?>> getInterfaces() {
313    SortedSet<Class<?>> interfaces = this.interfaces;
314    if (interfaces == null) {
315      if (this.isEmpty()) {
316        interfaces = Collections.emptySortedSet();
317      } else {
318        interfaces = new TreeSet<>(Types.typeNameComparator);
319        for (final Type type : this) {
320          addInterfaces(type, interfaces);
321        }
322        if (interfaces.isEmpty()) {
323          interfaces = Collections.emptySortedSet();
324        } else {
325          interfaces = Collections.unmodifiableSortedSet(interfaces);
326        }
327      }
328      this.interfaces = interfaces;
329    }
330    return interfaces;
331  }

334   * Returns a non-{@code null}, {@linkplain
335   * Collections#unmodifiableSet(Set) unmodifiable} {@link Set} of all
336   * {@link Type}s representing interfaces contained directly or
337   * indirectly by this {@link TypeSet}.
338   *
339   * <p>{@link Type}s contained directly or indirectly by this {@link
340   * TypeSet} that are not instances of either {@link Class} or {@link
341   * ParameterizedType} are deliberately ignored.</p>
342   *
343   * @return a non-{@code null}, {@linkplain
344   * Collections#unmodifiableSet(Set) unmodifiable} {@link Set} of all
345   * {@link Type}s representing interfaces contained directly or
346   * indirectly by this {@link TypeSet}
347   *
348   * @nullability This method never returns {@code null}.
349   *
350   * @threadsafety This method is safe for concurrent use by multiple
351   * threads.
352   *
353   * @idempotency This method is idempotent and deterministic.  The
354   * {@link Set} that is returned consists of elements in a
355   * particular, but deliberately unspecified, order that is not
356   * dependent upon the ordering of the {@link Set} of {@link Type}s
357   * initially supplied to this {@link TypeSet}.
358   */
359  public final SortedSet<Type> getGenericInterfaces() {
360    SortedSet<Type> interfaces = this.genericInterfaces;
361    if (interfaces == null) {
362      if (this.isEmpty()) {
363        interfaces = Collections.emptySortedSet();
364      } else {
365        interfaces = new TreeSet<>(Types.typeNameComparator);
366        for (final Type type : this) {
367          addGenericInterfaces(type, interfaces); // adds to interfaces
368        }
369        if (interfaces.isEmpty()) {
370          interfaces = Collections.emptySortedSet();
371        } else {
372          interfaces = Collections.unmodifiableSortedSet(interfaces);
373        }
374      }
375      this.genericInterfaces = interfaces;
376    }
377    return interfaces;
378  }

380  /**
381   * Returns an arbitrarily selected {@link Class} that is guaranteed
382   * to be the most specialized interface in its type hierarchy drawn
383   * from the {@linkplain #getInterfaces() interfaces} in this {@link
384   * TypeSet}, or {@code null} if this {@link TypeSet} contains no
385   * interfaces.
386   *
387   * <p>The {@link Class} that is returned is guaranteed to return
388   * {@code true} from its {@link Class#isInterface()} method.</p>
389   *
390   * <p>For example, if a {@link TypeSet} contains {@link
391   * java.io.Closeable Closeable.class} and {@link AutoCloseable
392   * AutoCloseable.class}, then this method will return {@link
393   * java.io.Closeable Closeable.class}.  If a {@link TypeSet}
394   * contains {@link AutoCloseable AutoCloseable.class} and {@link
395   * java.io.Serializable Serializable.class}, then either {@link
396   * AutoCloseable AutoCloseable.class} or {@link java.io.Serializable
397   * Serializable.class} will be returned.</p>
398   *
399   * @return an arbitrarily selected {@link Class} that is guaranteed
400   * to be the most specialized interface in its type hierarchy drawn
401   * from the {@linkplain #getInterfaces() interfaces} in this {@link
402   * TypeSet}, or {@code null}
403   *
404   * @nullability This method may return {@code null}.
405   *
406   * @idempotency This method is idempotent and deterministic.
407   *
408   * @threadsafety This method is safe for concurrent use by multiple
409   * threads.
410   *
411   * @see #getMostSpecializedClass()
412   */
413  public final Class<?> getMostSpecializedInterface() {
414    final Class<?> mostSpecializedInterface = this.mostSpecializedInterface;
415    final Class<?> returnValue;
416    if (mostSpecializedInterface == null) {
417      final Set<Class<?>> interfaces = this.getInterfaces();
418      if (interfaces == null || interfaces.isEmpty()) {
419        this.mostSpecializedInterface = Object.class; // marker
420        returnValue = null;
421      } else {
422        final Iterator<? extends Class<?>> interfacesIterator = interfaces.iterator();
423        Class<?> candidate = interfacesIterator.next();
424        while (interfacesIterator.hasNext()) {
425          Class<?> intrface = interfacesIterator.next();
426          if (candidate.isAssignableFrom(intrface)) {
427            candidate = intrface;
428          }
429        }
430        this.mostSpecializedInterface = returnValue = candidate;
431      }
432    } else if (mostSpecializedInterface.equals(Object.class)) {
433      // marker
434      returnValue = null;
435    } else {
436      returnValue = mostSpecializedInterface;
437    }
438    return returnValue;
439  }

441  /**
442   * Returns an arbitrarily selected {@link Class} that is guaranteed
443   * to be the most specialized subclass in its type hierarchy drawn
444   * from the {@linkplain #getClasses() classes} in this {@link
445   * TypeSet}.
446   *
447   * <p>For example, if a {@link TypeSet} contains {@link Integer
448   * Integer.class}, {@link Number Number.class} and {@link Object
449   * Object.class}, then this method will return {@link Integer
450   * Integer.class}.  If a {@link TypeSet} contains {@link Integer
451   * Integer.class} and {@link String String.class}, then either
452   * {@link Integer Integer.class} or {@link String String.class} will
453   * be returned.</p>
454   *
455   * <p>In practice, most {@link TypeSet}s in normal use contain
456   * classes from a single inheritance hierarchy.</p>
457   *
458   * @return an arbitrarily selected {@link Class} that is guaranteed
459   * to be the most specialized subclass in its type hierarchy drawn
460   * from the {@linkplain #getClasses() classes} in this {@link
461   * TypeSet}; never {@code null}
462   *
463   * @nullability This method never returns {@code null}.
464   *
465   * @idempotency This method is idempotent and deterministic.
466   *
467   * @threadsafety This method is safe for concurrent use by multiple
468   * threads.
469   *
470   * @see #getMostSpecializedInterface()
471   */
472  public final Class<?> getMostSpecializedClass() {
473    final Class<?> mostSpecializedClass = this.mostSpecializedClass;
474    final Class<?> returnValue;
475    if (mostSpecializedClass == null) {
476      final Set<Class<?>> classes = this.getClasses();
477      if (classes == null || classes.isEmpty()) {
478        this.mostSpecializedClass = returnValue = Object.class;
479      } else {
480        final Iterator<? extends Class<?>> classesIterator = classes.iterator();
481        assert classesIterator.hasNext();
482        Class<?> candidateClass = classesIterator.next();
483        assert candidateClass != null;
484        assert !candidateClass.isInterface();
485        while (classesIterator.hasNext()) {
486          final Class<?> cls = classesIterator.next();
487          assert cls != null;
488          assert !cls.isInterface();
489          if (candidateClass.isAssignableFrom(cls)) {
490            candidateClass = cls;
491          }
492        }
493        this.mostSpecializedClass = returnValue = candidateClass;
494      }
495    } else {
496      returnValue = mostSpecializedClass;
497    }
498    assert returnValue != null;
499    return returnValue;
500  }

633  private static final Type addClass(final Type type, final SortedSet<Type> set) {
634    final Type returnValue;
635    if (type == null) {
636      returnValue = null;
637    } else if (type instanceof Class) {
638      final Class<?> c = (Class<?>)type;
639      if (c.isInterface()) {
640        returnValue = null;
641      } else {
642        set.add(type);
643        returnValue = c.getGenericSuperclass();
644      }
645    } else if (type instanceof ParameterizedType) {
646      returnValue = addClass(((ParameterizedType)type).getRawType(), set); // XXX recursive
647    } else {
648      returnValue = null;
649    }
650    return returnValue;
651  }
652
653  private static final void addInterfaces(final Type type, final Set<Class<?>> interfaces) {
654    if (type == null) {
655      // do nothing
656    } else if (type instanceof Class) {
657      final Class<?> cls = (Class<?>)type;
658      final Type superclass;
659      if (cls.isInterface()) {
660        interfaces.add(cls);
661        superclass = null;
662      } else {
663        superclass = cls.getGenericSuperclass();
664      }
665      for (final Class<?> intrface : cls.getInterfaces()) {
666        if (interfaces.add(intrface)) {
667          addInterfaces(intrface, interfaces); // XXX recursive
668        }
669      }
670      if (superclass != null) {
671        addInterfaces(superclass, interfaces); // XXX recursive
672      }
673    } else if (type instanceof ParameterizedType) {
674      addInterfaces(((ParameterizedType)type).getRawType(), interfaces); // XXX recursive
675    } else {
676      // do nothing
677    }
678  }
679
680  private static final void addGenericInterfaces(final Type type, final Set<Type> interfaces) {
681    if (type == null) {
682      // do nothing
683    } else if (type instanceof Class) {
684      Class<?> c = (Class<?>)type;
685      final Type superclass;
686      if (c.isInterface()) {
687        interfaces.add(c);
688        superclass = null;
689      } else {
690        superclass = c.getGenericSuperclass();
691      }
692      for (final Type intrface : c.getGenericInterfaces()) {
693        if (interfaces.add(intrface)) {
694          addGenericInterfaces(intrface, interfaces); // XXX recursive
695        }
696      }
697      if (superclass != null) {
698        addGenericInterfaces(superclass, interfaces); // XXX recursive
699      }
700    } else if (type instanceof ParameterizedType) {
701      addGenericInterfaces(((ParameterizedType)type).getRawType(), interfaces); // XXX recursive
702    } else {
703      // do nothing
704    }
705  }
