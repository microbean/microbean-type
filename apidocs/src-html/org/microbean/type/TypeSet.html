031/**
032 * An {@linkplain Collections#unmodifiableSet(Set) unmodifiable}
033 * {@link AbstractSet} of {@link Type}s with additional convenient
034 * functionality.
035 *
036 * <p>{@linkplain #iterator() Iteration} over any {@link TypeSet}
037 * instance is guaranteed to be in some deterministic but otherwise
038 * unspecified order.  The order is subject to change between versions
039 * of this class.</p>
040 *
041 * @author <a href="https://about.me/lairdnelson"
042 * target="_parent">Laird Nelson</a>
043 */
044public class TypeSet extends AbstractSet<Type> {

097  /**
098   * Creates a {@link TypeSet} that {@linkplain #contains(Object)
099   * contains} only the supplied {@link Type} unless the supplied
100   * {@link Type} is {@code null}, in which case the resulting {@link
101   * TypeSet} will be forever {@linkplain #isEmpty() empty}.
102   *
103   * @param type the {@link Type} the new {@link TypeSet} will
104   * contain; may be {@code null} in which case the new {@link
105   * TypeSet} will be {@linkplain #isEmpty() empty}
106   */
107  public TypeSet(final Type type) {
108    this(type == null ? Collections.emptySet() : Collections.singleton(type));
109  }
110
111  /**
112   * Creates a {@link TypeSet} that {@linkplain #contains(Object)
113   * contains} only the {@link Type}s supplied unless the supplied
114   * {@link Collection} is {@code null} or {@linkplain
115   * Collection#isEmpty() empty}, in which case the resulting {@link
116   * TypeSet} will be forever {@linkplain #isEmpty() empty}.
117   *
118   * @param types the {@link Type}s the new {@link TypeSet} will
119   * contain; may be {@code null} or {@linkplain Collection#isEmpty()
120   * empty} in which case the new {@link TypeSet} will be forever
121   * {@linkplain #isEmpty() empty}
122   */
123  public TypeSet(final Collection<? extends Type> types) {
124    super();
125    if (types == null || types.isEmpty()) {
126      this.typeSet = Collections.emptySortedSet();
127    } else {
128      final SortedSet<Type> sortedTypeSet = new TreeSet<>(Types.typeNameComparator);
129      sortedTypeSet.addAll(types);
130      this.typeSet = Collections.unmodifiableSortedSet(sortedTypeSet);
131    }
132  }

140  /**
141   * Returns a non-{@code null}, {@linkplain
142   * Collections#unmodifiableSet(Set) unmodifiable} {@link SortedSet}
143   * of all raw types contained directly or indirectly by this {@link
144   * TypeSet}.
145   *
146   * <p>This method calls the {@link #getClasses()} and {@link
147   * #getInterfaces()} methods as part of its operation.</p>
148   *
149   * @return a non-{@code null}, {@linkplain
150   * Collections#unmodifiableSet(Set) unmodifiable} {@link SortedSet}
151   * of all raw types contained directly or indirectly by this {@link
152   * TypeSet}
153   *
154   * @see #getClasses()
155   *
156   * @see #getInterfaces()
157   *
158   * @nullability This method never returns {@code null}.
159   *
160   * @threadsafety This method is safe for concurrent use by multiple
161   * threads.
162   *
163   * @idempotency This method is idempotent and deterministic.  The
164   * {@link Set} that is returned consists of elements in a
165   * particular, but deliberately unspecified, order that is not
166   * dependent upon the ordering of the {@link Set} of {@link Type}s
167   * initially supplied to this {@link TypeSet}.
168   */
169  public final SortedSet<Class<?>> getRawTypes() {
170    SortedSet<Class<?>> rawTypes = this.rawTypes;
171    if (rawTypes == null) {
172      final SortedSet<Class<?>> classes = this.getClasses();
173      final SortedSet<Class<?>> interfaces = this.getInterfaces();
174      if (classes.isEmpty()) {
175        if (interfaces.isEmpty()) {
176          rawTypes = Collections.emptySortedSet();
177        } else {
178          rawTypes = interfaces;
179        }
180      } else if (interfaces.isEmpty()) {
181        rawTypes = classes;
182      } else {
183        final SortedSet<Class<?>> target = new TreeSet<>(Types.typeNameComparator);
184        target.addAll(classes);
185        target.addAll(interfaces);
186        rawTypes = Collections.unmodifiableSortedSet(target);
187      }
188      this.rawTypes = rawTypes;
189    }
190    return rawTypes;
191  }

193  /**
194   * Returns a non-{@code null}, {@linkplain
195   * Collections#unmodifiableSet(Set) unmodifiable} {@link SortedSet}
196   * of all {@link Class}es (not interfaces) contained directly or
197   * indirectly by this {@link TypeSet}.
198   *
199   * @return a non-{@code null}, {@linkplain
200   * Collections#unmodifiableSet(Set) unmodifiable} {@link Set} of all
201   * {@link Class}es (not interfaces) contained directly or indirectly
202   * by this {@link TypeSet}
203   *
204   * @nullability This method never returns {@code null}.
205   *
206   * @threadsafety This method is safe for concurrent use by multiple
207   * threads.
208   *
209   * @idempotency This method is idempotent and deterministic.  The
210   * {@link Set} that is returned consists of elements in a
211   * particular, but deliberately unspecified, order that is not
212   * dependent upon the ordering of the {@link Set} of {@link Type}s
213   * initially supplied to this {@link TypeSet}.
214   */
215  public final SortedSet<Class<?>> getClasses() {
216    SortedSet<Class<?>> classes = this.classes;
217    if (classes == null) {
218      if (this.isEmpty()) {
219        classes = Collections.emptySortedSet();
220      } else {
221        classes = new TreeSet<>(Types.typeNameComparator);
222        for (final Type type : this) {
223          // TODO: shouldn't we ignore any Type that is not either a Class or a ParameterizedType?
224          Class<?> rawType = Types.toClass(type);
225          while (rawType != null) {
226            if (!rawType.isInterface()) {
227              classes.add(rawType);
228            }
229            rawType = rawType.getSuperclass();
230          }
231        }
232        if (classes.isEmpty()) {
233          classes = Collections.emptySortedSet();
234        } else {
235          classes = Collections.unmodifiableSortedSet(classes);
236        }
237      }
238      this.classes = classes;
239    }
240    return classes;
241  }

243  /**
244   * Returns a non-{@code null}, {@linkplain
245   * Collections#unmodifiableSet(Set) unmodifiable} {@link SortedSet}
246   * of all {@link Type}s representing classes (not interfaces)
247   * contained directly or indirectly by this {@link TypeSet}.
248   *
249   * <p>{@link Type}s contained directly or indirectly by this {@link
250   * TypeSet} that are not instances of either {@link Class} or {@link
251   * ParameterizedType} are deliberately ignored.</p>
252   *
253   * @return a non-{@code null}, {@linkplain
254   * Collections#unmodifiableSet(Set) unmodifiable} {@link Set} of all
255   * {@link Type}s representing classes (not interfaces) contained
256   * directly or indirectly by this {@link TypeSet}
257   *
258   * @nullability This method never returns {@code null}.
259   *
260   * @threadsafety This method is safe for concurrent use by multiple
261   * threads.
262   *
263   * @idempotency This method is idempotent and deterministic.  The
264   * {@link Set} that is returned consists of elements in a
265   * particular, but deliberately unspecified, order that is not
266   * dependent upon the ordering of the {@link Set} of {@link Type}s
267   * initially supplied to this {@link TypeSet}.
268   */
269  public final SortedSet<Type> getGenericClasses() {
270    SortedSet<Type> genericClasses = this.genericClasses;
271    if (genericClasses == null) {
272      if (this.isEmpty()) {
273        genericClasses = Collections.emptySortedSet();
274      } else {
275        genericClasses = new TreeSet<>(Types.typeNameComparator);
276        for (Type type : this) {
277          while (type != null) {
278            type = addClass(type, genericClasses);
279          }
280        }
281        if (genericClasses.isEmpty()) {
282          genericClasses = Collections.emptySortedSet();
283        } else {
284          genericClasses = Collections.unmodifiableSortedSet(genericClasses);
285        }
286      }
287      this.genericClasses = genericClasses;
288    }
289    return genericClasses;
290  }

382  /**
383   * Returns an arbitrarily selected {@link Class} that is guaranteed
384   * to be the most specialized interface in its type hierarchy drawn
385   * from the {@linkplain #getInterfaces() interfaces} in this {@link
386   * TypeSet}, or {@code null} if this {@link TypeSet} contains no
387   * interfaces.
388   *
389   * <p>The {@link Class} that is returned is guaranteed to return
390   * {@code true} from its {@link Class#isInterface()} method.</p>
391   *
392   * <p>For example, if a {@link TypeSet} contains {@link
393   * java.io.Closeable Closeable.class} and {@link AutoCloseable
394   * AutoCloseable.class}, then this method will return {@link
395   * java.io.Closeable Closeable.class}.  If a {@link TypeSet}
396   * contains {@link AutoCloseable AutoCloseable.class} and {@link
397   * java.io.Serializable Serializable.class}, then either {@link
398   * AutoCloseable AutoCloseable.class} or {@link java.io.Serializable
399   * Serializable.class} will be returned.</p>
400   *
401   * @return an arbitrarily selected {@link Class} that is guaranteed
402   * to be the most specialized interface in its type hierarchy drawn
403   * from the {@linkplain #getInterfaces() interfaces} in this {@link
404   * TypeSet}, or {@code null}
405   *
406   * @nullability This method may return {@code null}.
407   *
408   * @idempotency This method is idempotent and deterministic.
409   *
410   * @threadsafety This method is safe for concurrent use by multiple
411   * threads.
412   *
413   * @see #getMostSpecializedClass()
414   */
415  public final Class<?> getMostSpecializedInterface() {
416    final Class<?> returnValue;
417    final Class<?> mostSpecializedInterface = this.mostSpecializedInterface;
418    if (mostSpecializedInterface == null) {
419      final Set<Class<?>> interfaces = this.getInterfaces();
420      if (interfaces == null || interfaces.isEmpty()) {
421        this.mostSpecializedInterface = Object.class; // marker
422        returnValue = null;
423      } else {
424        final Iterator<? extends Class<?>> interfacesIterator = interfaces.iterator();
425        Class<?> candidate = interfacesIterator.next();
426        while (interfacesIterator.hasNext()) {
427          Class<?> intrface = interfacesIterator.next();
428          if (candidate.isAssignableFrom(intrface)) {
429            candidate = intrface;
430          }
431        }
432        this.mostSpecializedInterface = returnValue = candidate;
433      }
434    } else if (mostSpecializedInterface.equals(Object.class)) {
435      // marker
436      returnValue = null;
437    } else {
438      returnValue = mostSpecializedInterface;
439    }
440    return returnValue;
441  }

507  /**
508   * Returns an arbitrarily selected {@link Class} that is guaranteed
509   * to be the most specialized subclass in its type hierarchy drawn
510   * from the {@linkplain #getClasses() classes} in this {@link
511   * TypeSet}.
512   *
513   * <p>For example, if a {@link TypeSet} contains {@link Integer
514   * Integer.class}, {@link Number Number.class} and {@link Object
515   * Object.class}, then this method will return {@link Integer
516   * Integer.class}.  If a {@link TypeSet} contains {@link Integer
517   * Integer.class} and {@link String String.class}, then either
518   * {@link Integer Integer.class} or {@link String String.class} will
519   * be returned.</p>
520   *
521   * <p>In practice, most {@link TypeSet}s in normal use contain
522   * classes from a single inheritance hierarchy.</p>
523   *
524   * @return an arbitrarily selected {@link Class} that is guaranteed
525   * to be the most specialized subclass in its type hierarchy drawn
526   * from the {@linkplain #getClasses() classes} in this {@link
527   * TypeSet}; never {@code null}
528   *
529   * @nullability This method never returns {@code null}.
530   *
531   * @idempotency This method is idempotent and deterministic.
532   *
533   * @threadsafety This method is safe for concurrent use by multiple
534   * threads.
535   *
536   * @see #getMostSpecializedInterface()
537   */
538  public final Class<?> getMostSpecializedClass() {
539    final Class<?> mostSpecializedClass = this.mostSpecializedClass;
540    final Class<?> returnValue;
541    if (mostSpecializedClass == null) {
542      final Set<Class<?>> classes = this.getClasses();
543      if (classes == null || classes.isEmpty()) {
544        this.mostSpecializedClass = returnValue = Object.class;
545      } else {
546        final Iterator<? extends Class<?>> classesIterator = classes.iterator();
547        Class<?> candidateClass = classesIterator.next();
548        while (classesIterator.hasNext()) {
549          final Class<?> cls = classesIterator.next();
550          if (candidateClass.isAssignableFrom(cls)) {
551            candidateClass = cls;
552          }
553        }
554        this.mostSpecializedClass = returnValue = candidateClass;
555      }
556    } else {
557      returnValue = mostSpecializedClass;
558    }
559    return returnValue;
560  }

750  private static final Type addClass(final Type type, final SortedSet<Type> set) {
751    final Type returnValue;
752    if (type == null) {
753      returnValue = null;
754    } else if (type instanceof Class) {
755      final Class<?> c = (Class<?>)type;
756      if (c.isInterface()) {
757        returnValue = null;
758      } else {
759        set.add(type);
760        returnValue = c.getGenericSuperclass();
761      }
762    } else if (type instanceof ParameterizedType) {
763      final Type rawType = ((ParameterizedType)type).getRawType();
764      if (rawType instanceof Class) {
765        final Class<?> c = (Class<?>)rawType;
766        if (c.isInterface()) {
767          returnValue = null;
768        } else {
769          set.add(type);
770          returnValue = c.getGenericSuperclass();
771        }
772      } else {
773        returnValue = null;
774      }
775    } else {
776      returnValue = null;
777    }
778    return returnValue;
779  }

781  private static final void addInterfaces(final Type type, final Set<Class<?>> interfaces) {
782    if (type == null) {
783      // do nothing
784    } else if (type instanceof Class) {
785      final Class<?> cls = (Class<?>)type;
786      final Type superclass;
787      if (cls.isInterface()) {
788        interfaces.add(cls);
789        superclass = null;
790      } else {
791        superclass = cls.getGenericSuperclass();
792      }
793      for (final Class<?> intrface : cls.getInterfaces()) {
794        if (interfaces.add(intrface)) {
795          addInterfaces(intrface, interfaces); // XXX recursive
796        }
797      }
798      if (superclass != null) {
799        addInterfaces(superclass, interfaces); // XXX recursive
800      }
801    } else if (type instanceof ParameterizedType) {
802      addInterfaces(((ParameterizedType)type).getRawType(), interfaces); // XXX recursive
803    } else {
804      // do nothing
805    }
806  }

855  private static final boolean isAssignableFrom(final Type receiverType, final Type payloadType) {
856    if (receiverType == null || payloadType == null) {
857      return false;
858    } else if (receiverType instanceof Class) {
859      if (payloadType instanceof Class) {
860        return ((Class<?>)receiverType).isAssignableFrom((Class<?>)payloadType);
861      } else if (payloadType instanceof ParameterizedType) {
862        return isAssignableFrom(receiverType, ((ParameterizedType)payloadType).getRawType());
863      } else {
864        throw new IllegalArgumentException("Unexpected or unhandled payloadType: " + payloadType);
865      }
866    } else if (receiverType instanceof ParameterizedType) {
867      return isAssignableFrom(((ParameterizedType)receiverType).getRawType(), payloadType);
868    } else {
869      throw new IllegalArgumentException("Unexpected or unhandled receiverType: " + receiverType);
870    }
871  }
